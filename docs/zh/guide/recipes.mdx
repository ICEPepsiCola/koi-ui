# 场景示例

可直接抄的成品切片。预览工具条可切换主题 / 桌面 / 移动。

## 登录

```tsx preview
import {
  Button,
  Card,
  Checkbox,
  Form,
  Input,
  Space,
  Stack,
  Text,
  toast,
} from '@koi-ui/core';

export default () => (
  <Card className="max-w-md w-full">
    <Stack gap={4}>
      <div>
        <Text className="text-lg font-semibold">欢迎回来</Text>
        <Text className="text-sm text-muted-foreground">登录后继续使用</Text>
      </div>
      <Form
        layout="vertical"
        onFinish={() => toast.success('登录成功')}
        onFinishFailed={() => toast.error('请检查表单')}
      >
        <Form.Item
          name="email"
          label="邮箱"
          rules={[
            { required: true, message: '请输入邮箱' },
            { type: 'email', message: '邮箱格式不正确' },
          ]}
        >
          <Input placeholder="name@company.com" clearable />
        </Form.Item>
        <Form.Item
          name="password"
          label="密码"
          rules={[{ required: true, min: 6, message: '至少 6 位' }]}
        >
          <Input type="password" placeholder="请输入密码" clearable />
        </Form.Item>
        <Form.Item name="remember" valuePropName="checked">
          <Checkbox label="记住我" />
        </Form.Item>
        <Space>
          <Button type="submit" variant="primary">登录</Button>
          <Button type="button" variant="ghost">注册账号</Button>
        </Space>
      </Form>
    </Stack>
  </Card>
);
```

## 数据列表

```tsx preview
import { useState } from 'react';
import {
  Badge,
  Button,
  SearchBar,
  Space,
  Stack,
  Table,
  Tag,
} from '@koi-ui/core';

const rows = [
  { id: '1', name: '北港团队', status: 'active', owners: 12 },
  { id: '2', name: '雪松实验室', status: 'trial', owners: 4 },
  { id: '3', name: '月光公司', status: 'paused', owners: 7 },
];

export default () => {
  const [query, setQuery] = useState('');
  const data = rows.filter((row) => row.name.includes(query.trim()));

  return (
    <Stack gap={4} className="w-full">
      <Space className="w-full justify-between" wrap>
        <SearchBar value={query} onChange={setQuery} placeholder="筛选团队" />
        <Button variant="soft">新建团队</Button>
      </Space>
      <Table
        columns={[
          { key: 'name', title: '团队' },
          {
            key: 'status',
            title: '状态',
            render: (value) =>
              value === 'active' ? (
                <Tag variant="success">进行中</Tag>
              ) : value === 'trial' ? (
                <Tag variant="warning">试用</Tag>
              ) : (
                <Tag variant="ghost">已暂停</Tag>
              ),
          },
          {
            key: 'owners',
            title: '成员',
            render: (value) => <Badge count={Number(value)} variant="soft" />,
          },
        ]}
        data={data}
      />
    </Stack>
  );
};
```

## 设置表单

```tsx preview
import {
  Button,
  Form,
  Input,
  Select,
  Space,
  Stack,
  Switch,
  Text,
  toast,
} from '@koi-ui/core';

export default () => {
  const [form] = Form.useForm();
  return (
    <Stack gap={4} className="max-w-lg w-full">
      <div>
        <Text className="text-lg font-semibold">工作区设置</Text>
        <Text className="text-sm text-muted-foreground">
          Form 自适应：移动端标签自动纵向排列。
        </Text>
      </div>
      <Form
        form={form}
        initialValues={{ name: 'Koi 工作室', plan: 'pro', notify: true }}
        onFinish={() => toast.success('已保存')}
      >
        <Form.Item
          name="name"
          label="工作区名称"
          rules={[{ required: true, message: '必填' }]}
        >
          <Input clearable />
        </Form.Item>
        <Form.Item name="plan" label="套餐">
          <Select
            options={[
              { value: 'free', label: '免费' },
              { value: 'pro', label: '专业版' },
              { value: 'team', label: '团队版' },
            ]}
          />
        </Form.Item>
        <Form.Item name="notify" label="邮件提醒" valuePropName="checked">
          <Switch />
        </Form.Item>
        <Space>
          <Button type="submit" variant="primary">保存</Button>
          <Button type="button" variant="outline" onClick={() => form.resetFields()}>
            重置
          </Button>
        </Space>
      </Form>
    </Stack>
  );
};
```
