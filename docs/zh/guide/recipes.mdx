# 场景示例

可直接抄的成品切片。预览工具条可切换主题 / 桌面 / 移动。

## 登录

```tsx preview
import { useState } from 'react';
import {
  Alert,
  Button,
  Card,
  Checkbox,
  Form,
  FormItem,
  Input,
  Space,
  Stack,
  Text,
} from '@koi-ui/core';

export default () => {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [error, setError] = useState('');

  return (
    <Card className="max-w-md w-full">
      <Stack gap={4}>
        <div>
          <Text className="text-lg font-semibold">欢迎回来</Text>
          <Text className="text-sm text-muted-foreground">登录后继续使用</Text>
        </div>
        {error ? <Alert variant="error" appearance="soft" title={error} /> : null}
        <Form layout="vertical">
          <FormItem label="邮箱" required>
            <Input
              value={email}
              onChange={setEmail}
              placeholder="name@company.com"
              clearable
            />
          </FormItem>
          <FormItem label="密码" required>
            <Input
              type="password"
              value={password}
              onChange={setPassword}
              placeholder="请输入密码"
              clearable
            />
          </FormItem>
          <Checkbox label="记住我" />
          <Space>
            <Button
              variant="primary"
              onClick={() =>
                setError(email && password ? '' : '请填写邮箱和密码')
              }
            >
              登录
            </Button>
            <Button variant="ghost">注册账号</Button>
          </Space>
        </Form>
      </Stack>
    </Card>
  );
};
```

## 数据列表

```tsx preview
import { useState } from 'react';
import {
  Badge,
  Button,
  SearchBar,
  Space,
  Stack,
  Table,
  Tag,
} from '@koi-ui/core';

const rows = [
  { id: '1', name: '北港团队', status: 'active', owners: 12 },
  { id: '2', name: '雪松实验室', status: 'trial', owners: 4 },
  { id: '3', name: '月光公司', status: 'paused', owners: 7 },
];

export default () => {
  const [query, setQuery] = useState('');
  const data = rows.filter((row) => row.name.includes(query.trim()));

  return (
    <Stack gap={4} className="w-full">
      <Space className="w-full justify-between" wrap>
        <SearchBar value={query} onChange={setQuery} placeholder="筛选团队" />
        <Button variant="soft">新建团队</Button>
      </Space>
      <Table
        columns={[
          { key: 'name', title: '团队' },
          {
            key: 'status',
            title: '状态',
            render: (value) =>
              value === 'active' ? (
                <Tag variant="success">进行中</Tag>
              ) : value === 'trial' ? (
                <Tag variant="warning">试用</Tag>
              ) : (
                <Tag variant="ghost">已暂停</Tag>
              ),
          },
          {
            key: 'owners',
            title: '成员',
            render: (value) => <Badge count={Number(value)} variant="soft" />,
          },
        ]}
        data={data}
      />
    </Stack>
  );
};
```

## 设置表单

```tsx preview
import { useState } from 'react';
import {
  Alert,
  Button,
  Form,
  FormItem,
  Input,
  Select,
  Space,
  Stack,
  Switch,
  Text,
} from '@koi-ui/core';

export default () => {
  const [name, setName] = useState('Koi 工作室');
  const [plan, setPlan] = useState('pro');
  const [notify, setNotify] = useState(true);
  const [saved, setSaved] = useState(false);

  return (
    <Stack gap={4} className="max-w-lg w-full">
      <div>
        <Text className="text-lg font-semibold">工作区设置</Text>
        <Text className="text-sm text-muted-foreground">
          Form 自适应：移动端标签自动纵向排列。
        </Text>
      </div>
      {saved ? (
        <Alert variant="success" appearance="soft" title="已保存" description="设置已更新。" />
      ) : null}
      <Form>
        <FormItem label="工作区名称" required>
          <Input value={name} onChange={setName} clearable />
        </FormItem>
        <FormItem label="套餐">
          <Select
            value={plan}
            onChange={setPlan}
            options={[
              { value: 'free', label: '免费' },
              { value: 'pro', label: '专业版' },
              { value: 'team', label: '团队版' },
            ]}
          />
        </FormItem>
        <FormItem label="邮件提醒">
          <Switch checked={notify} onChange={setNotify} />
        </FormItem>
        <Space>
          <Button variant="primary" onClick={() => setSaved(true)}>保存</Button>
          <Button variant="outline" onClick={() => setSaved(false)}>重置</Button>
        </Space>
      </Form>
    </Stack>
  );
};
```
