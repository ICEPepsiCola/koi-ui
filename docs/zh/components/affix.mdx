# Affix 固钉

## 基础用法

```tsx preview
import { Affix, Button } from '@koi-ui/core';
export default () => (
  <div className="space-y-3 pb-8 text-sm text-muted-foreground">
    <p className="font-medium text-surface-foreground">向下滚动查看固钉效果</p>
    <Affix offsetTop={0}>
      <Button>固定在顶部</Button>
    </Affix>
    {Array.from({ length: 40 }, (_, i) => (
      <p key={i}>内容行 {i + 1}</p>
    ))}
  </div>
);
```

## 底部固钉

```tsx preview
import { Affix, Button } from '@koi-ui/core';
export default () => (
  <div className="space-y-3 pb-8 text-sm text-muted-foreground">
    <p className="font-medium text-surface-foreground">向下滚动，底部按钮会吸底</p>
    {Array.from({ length: 40 }, (_, i) => (
      <p key={i}>内容行 {i + 1}</p>
    ))}
    <Affix offsetBottom={0}>
      <Button variant="primary">固定在底部</Button>
    </Affix>
  </div>
);
```
