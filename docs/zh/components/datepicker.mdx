# DatePicker 日期选择

## 基础用法

```tsx preview
import { DatePicker } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [v,setV]=useState(''); return <DatePicker value={v} onChange={setV} placeholder="选择日期" />; };
```

## 受控日期

```tsx preview
import { DatePicker, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [date, setDate] = useState('2026-07-12');
  return (
    <div>
      <DatePicker value={date} onChange={setDate} placeholder="选择日期" />
      <Text muted className="mt-2 block text-sm">选中：{date || '未选择'}</Text>
    </div>
  );
};
```

## clearable

```tsx preview
import { DatePicker } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [v,setV]=useState('2026-07-12'); return <DatePicker value={v} onChange={setV} clearable placeholder="选择日期" />; };
```
