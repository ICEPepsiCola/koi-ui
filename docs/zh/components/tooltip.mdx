# Tooltip 文字提示

## 基础用法

```tsx preview
import { Tooltip, Button } from '@koi-ui/core';
export default () => (
  <div className="flex justify-center pt-12">
    <Tooltip content="提示文字">
      <Button>悬停</Button>
    </Tooltip>
  </div>
);
```

## 位置

```tsx preview
import { Tooltip, Button } from '@koi-ui/core';

const cell = 'flex h-24 items-center justify-center';

export default () => (
  <div className="grid grid-cols-2 gap-4 sm:grid-cols-4">
    <div className={cell}>
      <Tooltip placement="top" content="上方" open>
        <Button size="sm">top</Button>
      </Tooltip>
    </div>
    <div className={cell}>
      <Tooltip placement="bottom" content="下方" open>
        <Button size="sm">bottom</Button>
      </Tooltip>
    </div>
    <div className={cell}>
      <Tooltip placement="left" content="左侧" open>
        <Button size="sm">left</Button>
      </Tooltip>
    </div>
    <div className={cell}>
      <Tooltip placement="right" content="右侧" open>
        <Button size="sm">right</Button>
      </Tooltip>
    </div>
  </div>
);
```

## 颜色

与 Button 同一套语义 `color`，默认 `neutral`。

```tsx preview
import { Tooltip, Button } from '@koi-ui/core';

const colors = [
  'neutral',
  'primary',
  'secondary',
  'info',
  'success',
  'warning',
  'error',
] as const;

export default () => (
  <div className="flex flex-wrap gap-x-3 gap-y-2">
    {colors.map((color) => (
      <div key={color} className="flex h-20 items-end">
        <Tooltip content={color} color={color} open>
          <Button color={color} size="sm">
            {color}
          </Button>
        </Tooltip>
      </div>
    ))}
  </div>
);
```
