# Form 表单

## 基础用法

```tsx preview
import { Form, Input, Button, toast } from '@koi-ui/core';

export default () => (
  <Form
    layout="vertical"
    onFinish={(values) => toast.success(`Hello ${values.username}`)}
  >
    <Form.Item
      name="username"
      label="用户名"
      rules={[{ required: true, message: '请输入用户名' }]}
    >
      <Input placeholder="请输入" clearable />
    </Form.Item>
    <Button type="submit" variant="primary">提交</Button>
  </Form>
);
```

## 校验与表单实例

```tsx preview
import { Form, Input, InputNumber, Switch, Button, Space, toast } from '@koi-ui/core';

export default () => {
  const [form] = Form.useForm();
  return (
    <Form
      form={form}
      layout="vertical"
      initialValues={{ age: 18, notify: true }}
      onFinish={(values) => toast.success(JSON.stringify(values))}
      onFinishFailed={() => toast.error('请检查表单')}
    >
      <Form.Item
        name="email"
        label="邮箱"
        rules={[
          { required: true, message: '必填' },
          { type: 'email', message: '邮箱格式不正确' },
        ]}
      >
        <Input placeholder="name@example.com" clearable />
      </Form.Item>
      <Form.Item name="age" label="年龄" rules={[{ type: 'number', min: 1, max: 120, message: '1-120' }]}>
        <InputNumber />
      </Form.Item>
      <Form.Item name="notify" label="通知" valuePropName="checked">
        <Switch />
      </Form.Item>
      <Space>
        <Button type="submit" variant="primary">提交</Button>
        <Button type="button" variant="outline" onClick={() => form.resetFields()}>重置</Button>
      </Space>
    </Form>
  );
};
```

## 水平布局

```tsx preview
import { Form, Input, Button, Space } from '@koi-ui/core';

export default () => (
  <Form
    layout="horizontal"
    onFinish={() => {}}
  >
    <Form.Item name="email" label="邮箱" rules={[{ required: true, message: '必填' }]}>
      <Input placeholder="name@example.com" />
    </Form.Item>
    <Form.Item name="password" label="密码" rules={[{ required: true, min: 6, message: '至少 6 位' }]}>
      <Input type="password" placeholder="请输入密码" />
    </Form.Item>
    <Space>
      <Button type="submit" variant="primary">登录</Button>
      <Button type="button" variant="secondary">注册</Button>
    </Space>
  </Form>
);
```

## 统一控件尺寸

```tsx preview
import {
  AutoComplete,
  Cascader,
  Input,
  SearchBar,
  Select,
  Stack,
  Text,
} from '@koi-ui/core';

const regionOptions = [
  { label: '浙江', value: 'zj', children: [{ label: '杭州', value: 'hz' }] },
];

export default () => (
  <Stack gap={3} className="max-w-md w-full">
    <Text muted size="sm">默认 md — 单行表单控件统一 32px 高度（同 Ant Design medium）</Text>
    <Input placeholder="输入框" />
    <Select
      placeholder="选择城市"
      options={[
        { label: '北京', value: 'bj' },
        { label: '上海', value: 'sh' },
      ]}
    />
    <Cascader options={regionOptions} placeholder="选择地区" />
    <SearchBar placeholder="搜索" />
    <AutoComplete
      placeholder="自动完成"
      options={[
        { value: 'apple', label: 'Apple' },
        { value: 'banana', label: 'Banana' },
      ]}
    />
    <Text muted size="sm" className="pt-2">size="sm" — 24px</Text>
    <Input size="sm" placeholder="输入框" />
    <Select
      size="sm"
      placeholder="选择城市"
      options={[{ label: '北京', value: 'bj' }, { label: '上海', value: 'sh' }]}
    />
    <Cascader size="sm" options={regionOptions} placeholder="选择地区" />
    <SearchBar size="sm" placeholder="搜索" />
  </Stack>
);
```
