# Popover 气泡卡片

## 基础用法

```tsx preview
import { Popover, Button } from '@koi-ui/core';
export default () => (
  <Popover content="气泡内容">
    <Button>点击查看</Button>
  </Popover>
);
```

## 标题

```tsx preview
import { Popover, Button } from '@koi-ui/core';
export default () => (
  <Popover title="标题" content="带标题的气泡卡片内容。">
    <Button variant="outline">带标题</Button>
  </Popover>
);
```

## 位置

```tsx preview
import { Popover, Button, Space } from '@koi-ui/core';
export default () => (
  <Space wrap>
    <Popover placement="top" content="上方">
      <Button size="sm">top</Button>
    </Popover>
    <Popover placement="bottom" content="下方">
      <Button size="sm">bottom</Button>
    </Popover>
    <Popover placement="left" content="左侧">
      <Button size="sm">left</Button>
    </Popover>
    <Popover placement="right" content="右侧">
      <Button size="sm">right</Button>
    </Popover>
  </Space>
);
```

## 悬停触发

```tsx preview
import { Popover, Button } from '@koi-ui/core';
export default () => (
  <Popover trigger="hover" content="鼠标悬停显示">
    <Button variant="soft">悬停触发</Button>
  </Popover>
);
```
