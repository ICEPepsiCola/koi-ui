# NavBar 导航栏

## 基础用法

```tsx preview
import { NavBar } from '@koi-ui/core';
export default () => <NavBar title="页面标题" onBack={() => {}} />;
```

## 左侧与返回图标

```tsx preview
import { NavBar, Button, Icon, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <NavBar
      title="自定义返回图标"
      onBack={() => {}}
      backIcon={<Icon name="close" size="md" />}
    />
    <NavBar
      title="自定义左侧"
      left={<Button size="sm" variant="ghost">关闭</Button>}
      right={<Button size="sm" variant="ghost">分享</Button>}
    />
  </Space>
);
```

## 固定、边框与安全区

```tsx preview
import { NavBar, Space, Text } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <div>
      <Text muted size="sm" className="mb-2 block">fixed=false · bordered</Text>
      <NavBar fixed={false} bordered title="相对定位" onBack={() => {}} />
    </div>
    <div>
      <Text muted size="sm" className="mb-2 block">bordered=false · safeArea=false</Text>
      <NavBar fixed={false} bordered={false} safeArea={false} title="无底边 / 无安全区" onBack={() => {}} />
    </div>
  </Space>
);
```
