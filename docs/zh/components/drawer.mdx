# Drawer 抽屉

## 基础用法

```tsx preview
import { Drawer, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [o,setO]=useState(false); return <><Button onClick={()=>setO(true)}>打开抽屉</Button><Drawer open={o} onClose={()=>setO(false)} title="抽屉标题">内容</Drawer></>; };
```

## 底部抽屉

```tsx preview
import { Drawer, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>底部弹出</Button>
      <Drawer open={open} onClose={() => setOpen(false)} title="筛选" placement="bottom">
        筛选面板内容
      </Drawer>
    </>
  );
};
```
