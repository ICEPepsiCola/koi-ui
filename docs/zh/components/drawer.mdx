# Drawer 抽屉

侧栏 / 底栏面板，chrome 与 Modal 对齐（无分割线、可 `closable`）。

## 基础用法

```tsx preview
import { Drawer, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>打开抽屉</Button>
      <Drawer open={open} onClose={() => setOpen(false)} title="抽屉标题" closable>
        内容
      </Drawer>
    </>
  );
};
```

## 弹出方向

```tsx preview
import { Drawer, Button, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [placement, setPlacement] = useState<'left' | 'right' | 'top' | 'bottom'>('right');
  const [open, setOpen] = useState(false);
  return (
    <>
      <Space wrap>
        {(['left', 'right', 'top', 'bottom'] as const).map((p) => (
          <Button
            key={p}
            size="sm"
            variant={placement === p ? 'solid' : 'outline'}
            onClick={() => {
              setPlacement(p);
              setOpen(true);
            }}
          >
            {p}
          </Button>
        ))}
      </Space>
      <Drawer
        open={open}
        onClose={() => setOpen(false)}
        title={`placement=${placement}`}
        placement={placement}
        closable
      >
        从 {placement} 方向滑出
      </Drawer>
    </>
  );
};
```

## 尺寸

```tsx preview
import { Drawer, Button, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [size, setSize] = useState<'sm' | 'md' | 'lg'>('md');
  const [open, setOpen] = useState(false);
  return (
    <>
      <Space>
        {(['sm', 'md', 'lg'] as const).map((s) => (
          <Button
            key={s}
            size="sm"
            variant={size === s ? 'solid' : 'outline'}
            onClick={() => {
              setSize(s);
              setOpen(true);
            }}
          >
            {s}
          </Button>
        ))}
      </Space>
      <Drawer
        open={open}
        onClose={() => setOpen(false)}
        title={`size=${size}`}
        size={size}
        closable
      >
        不同尺寸的抽屉宽度/高度
      </Drawer>
    </>
  );
};
```

## 底部操作

```tsx preview
import { Drawer, Button, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>带底部</Button>
      <Drawer
        open={open}
        onClose={() => setOpen(false)}
        title="编辑"
        closable
        footer={
          <Space>
            <Button color="neutral" variant="ghost" onClick={() => setOpen(false)}>
              取消
            </Button>
            <Button color="primary" onClick={() => setOpen(false)}>
              保存
            </Button>
          </Space>
        }
      >
        带 footer 的抽屉内容
      </Drawer>
    </>
  );
};
```
