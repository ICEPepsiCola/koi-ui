# Tag 标签

## 基础用法

```tsx preview
import { Tag, Space } from '@koi-ui/core';
export default () => (
  <Space wrap>
    <Tag>默认</Tag>
    <Tag color="secondary">次要</Tag>
    <Tag color="info">信息</Tag>
    <Tag color="success">成功</Tag>
    <Tag color="warning">警告</Tag>
    <Tag color="error">危险</Tag>
  </Space>
);
```

## 样式变体

```tsx preview
import { Tag, Space } from '@koi-ui/core';
export default () => (
  <Space wrap>
    <Tag color="primary" variant="soft">Soft</Tag>
    <Tag color="primary" variant="outline">Outline</Tag>
    <Tag color="primary" variant="dash">Dash</Tag>
    <Tag color="primary" variant="solid">Solid</Tag>
    <Tag color="primary" variant="ghost">Ghost</Tag>
  </Space>
);
```

## 可关闭标签

```tsx preview
import { Tag, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [tags, setTags] = useState(['React', 'TypeScript', 'Tailwind']);
  return (
    <Space wrap>
      {tags.map((tag) => (
        <Tag key={tag} closable onClose={() => setTags(tags.filter((t) => t !== tag))}>{tag}</Tag>
      ))}
    </Space>
  );
};
```
