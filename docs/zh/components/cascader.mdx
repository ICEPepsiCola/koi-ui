## 禁用

```tsx preview
import { Cascader } from '@koi-ui/core';
const options = [
  {
    label: '浙江',
    value: 'zj',
    children: [
      {
        label: '杭州',
        value: 'hz',
        children: [
          { label: '西湖', value: 'xh' },
          { label: '余杭', value: 'yh' },
        ],
      },
    ],
  },
];
export default () => (
  <Cascader
    disabled
    value={['zj', 'hz', 'xh']}
    options={options}
    placeholder="选择地区"
  />
);
```
