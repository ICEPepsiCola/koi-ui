
# Cascader 级联选择

## 基础用法

```tsx preview
import { Cascader } from '@koi-ui/core';
export default () => <Cascader options={[{label:'浙江',value:'zj',children:[{label:'杭州',value:'hz'}]}]} placeholder="选择地区" />;
```

## clearable

```tsx preview
import { Cascader } from '@koi-ui/core';
import { useState } from 'react';
const options = [{ label:'浙江', value:'zj', children:[{ label:'杭州', value:'hz' }] }];
export default () => {
  const [value, setValue] = useState(['zj', 'hz']);
  return <Cascader value={value} onChange={setValue} options={options} clearable placeholder="选择地区" />;
};
```
