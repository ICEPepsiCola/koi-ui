# ActionSheet 动作面板

## 基础用法

```tsx preview
import { ActionSheet, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>打开</Button>
      <ActionSheet
        open={open}
        onClose={() => setOpen(false)}
        actions={[
          { key: '1', text: '选项一' },
          { key: '2', text: '选项二' },
        ]}
      />
    </>
  );
};
```

## 带标题描述

```tsx preview
import { ActionSheet, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>更多操作</Button>
      <ActionSheet
        open={open}
        onClose={() => setOpen(false)}
        title="分享到"
        description="选择一个平台"
        actions={[
          { key: 'wechat', text: '微信' },
          { key: 'link', text: '复制链接' },
        ]}
      />
    </>
  );
};
```

## 危险与禁用

```tsx preview
import { ActionSheet, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button color="error" onClick={() => setOpen(true)}>
        管理
      </Button>
      <ActionSheet
        open={open}
        onClose={() => setOpen(false)}
        title="操作"
        cancelText="关闭"
        actions={[
          { key: 'edit', text: '编辑' },
          { key: 'archive', text: '归档', disabled: true },
          { key: 'delete', text: '删除', destructive: true },
        ]}
      />
    </>
  );
};
```
