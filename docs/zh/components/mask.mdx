# Mask 背景蒙层

## 基础用法

```tsx preview
import { Mask, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [o,setO]=useState(false); return <><Button onClick={()=>setO(true)}>显示蒙层</Button><Mask open={o} onClick={()=>setO(false)} /></>; };
```
