
# Table 表格

## 基础用法

```tsx preview
import { Table } from '@koi-ui/core';
export default () => <Table columns={[{key:'name',title:'姓名'},{key:'age',title:'年龄'}]} data={[{name:'张三',age:'20'},{name:'李四',age:'25'}]} />;
```

## 空数据

```tsx preview
import { Table } from '@koi-ui/core';
export default () => (
  <Table
    columns={[{ key: 'name', title: '姓名' }, { key: 'role', title: '角色' }]}
    data={[]}
    emptyText="暂无成员"
  />
);
```
