# Table 表格

## 基础用法

```tsx preview
import { Table } from '@koi-ui/core';
export default () => (
  <Table
    columns={[
      { key: 'name', title: '姓名' },
      { key: 'age', title: '年龄' },
    ]}
    data={[
      { name: '张三', age: '20' },
      { name: '李四', age: '25' },
    ]}
  />
);
```

## 行点击

```tsx preview
import { Table, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [clicked, setClicked] = useState('');
  return (
    <div>
      <Table
        columns={[
          { key: 'name', title: '姓名' },
          { key: 'role', title: '角色' },
        ]}
        data={[
          { name: '张三', role: '设计' },
          { name: '李四', role: '开发' },
        ]}
        onRowClick={(row) => setClicked(String(row.name))}
      />
      <Text muted className="mt-2 block text-sm">点击行：{clicked || '无'}</Text>
    </div>
  );
};
```

## 移动端字段

```tsx preview
import { Table } from '@koi-ui/core';
export default () => (
  <Table
    columns={[
      { key: 'name', title: '姓名' },
      { key: 'role', title: '角色' },
      { key: 'city', title: '城市' },
    ]}
    data={[
      { name: '张三', role: '设计', city: '杭州' },
      { name: '李四', role: '开发', city: '上海' },
    ]}
    mobileFields={['name', 'role']}
  />
);
```
