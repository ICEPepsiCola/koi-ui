# Popup 弹出层

## 基础用法

```tsx preview
import { Popup, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>打开弹出层</Button>
      <Popup open={open} onClose={() => setOpen(false)} title="提示">
        居中弹出的内容区域。
      </Popup>
    </>
  );
};
```

## 底部操作

```tsx preview
import { Popup, Button, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>带底部</Button>
      <Popup
        open={open}
        onClose={() => setOpen(false)}
        title="确认"
        footer={
          <div className="flex justify-end gap-2 p-3">
            <Space>
              <Button variant="secondary" size="sm" onClick={() => setOpen(false)}>
                取消
              </Button>
              <Button size="sm" onClick={() => setOpen(false)}>
                确定
              </Button>
            </Space>
          </div>
        }
      >
        确定继续吗？
      </Popup>
    </>
  );
};
```

## 点击遮罩不关闭

```tsx preview
import { Popup, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>遮罩不关闭</Button>
      <Popup
        open={open}
        onClose={() => setOpen(false)}
        title="重要提示"
        closeOnMask={false}
        footer={
          <div className="flex justify-end p-3">
            <Button size="sm" onClick={() => setOpen(false)}>
              我知道了
            </Button>
          </div>
        }
      >
        closeOnMask 为 false 时需通过按钮关闭。
      </Popup>
    </>
  );
};
```
