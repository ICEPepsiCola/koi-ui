# Popup 弹出层

紧凑居中对话框，样式与 Modal 同源，默认 `size="sm"`。

## 基础用法

```tsx preview
import { Popup, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>打开弹出层</Button>
      <Popup open={open} onClose={() => setOpen(false)} title="提示">
        居中弹出的内容区域。
      </Popup>
    </>
  );
};
```

## 操作区与关闭按钮

```tsx preview
import { Popup, Button, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>带底部</Button>
      <Popup
        open={open}
        onClose={() => setOpen(false)}
        title="确认"
        closable
        footer={
          <Space>
            <Button color="neutral" variant="ghost" size="sm" onClick={() => setOpen(false)}>
              取消
            </Button>
            <Button color="primary" size="sm" onClick={() => setOpen(false)}>
              确定
            </Button>
          </Space>
        }
      >
        确定继续吗？
      </Popup>
    </>
  );
};
```

## 点击遮罩不关闭

```tsx preview
import { Popup, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>遮罩不关闭</Button>
      <Popup
        open={open}
        onClose={() => setOpen(false)}
        title="重要提示"
        maskClosable={false}
        closable
        footer={
          <Button size="sm" onClick={() => setOpen(false)}>
            我知道了
          </Button>
        }
      >
        maskClosable 为 false 时需通过按钮关闭。
      </Popup>
    </>
  );
};
```
