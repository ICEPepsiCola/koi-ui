# Modal 对话框

## 基础用法

```tsx preview
import { Modal, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>打开弹窗</Button>
      <Modal open={open} onClose={() => setOpen(false)} title="标题">
        内容
      </Modal>
    </>
  );
};
```

## 带底部操作

```tsx preview
import { Modal, Button, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>确认操作</Button>
      <Modal
        open={open}
        onClose={() => setOpen(false)}
        title="删除确认"
        footer={
          <Space>
            <Button color="secondary" onClick={() => setOpen(false)}>
              取消
            </Button>
            <Button color="error" onClick={() => setOpen(false)}>
              删除
            </Button>
          </Space>
        }
      >
        确定要删除这条记录吗？
      </Modal>
    </>
  );
};
```

## 移动端高度

```tsx preview
import { Modal, Button, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [tall, setTall] = useState(false);
  const [open, setOpen] = useState(false);
  return (
    <>
      <Space>
        <Button onClick={() => {
            setTall(true);
            setOpen(true);
          }}
        >
          接近全屏
        </Button>
        <Button variant="outline"onClick={() => {
            setTall(false);
            setOpen(true);
          }}
        >
          较低高度
        </Button>
      </Space>
      <Modal
        open={open}
        onClose={() => setOpen(false)}
        title="移动端抽屉高度"
        mobileFullscreen={tall}
      >
        移动端以底部抽屉呈现；mobileFullscreen 为 true 时最高约 90vh，为 false 时约 70vh。
      </Modal>
    </>
  );
};
```
