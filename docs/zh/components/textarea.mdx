# TextArea 文本域

## 基础用法

```tsx preview
import { TextArea } from '@koi-ui/core';
export default () => <TextArea placeholder="请输入多行文本" rows={3} />;
```

## clearable

```tsx preview
import { TextArea } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [value, setValue] = useState('这段内容可以一键清除');
  return <TextArea value={value} onChange={setValue} clearable placeholder="请输入多行文本" rows={3} />;
};
```
