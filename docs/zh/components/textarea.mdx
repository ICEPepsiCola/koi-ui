# TextArea 文本域

## 基础用法

```tsx preview
import { TextArea } from '@koi-ui/core';
export default () => <TextArea placeholder="请输入多行文本" rows={3} />;
```

## 尺寸

```tsx preview
import { TextArea, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <TextArea size="sm" placeholder="sm" rows={2} />
    <TextArea size="md" placeholder="md" rows={3} />
    <TextArea size="lg" placeholder="lg" rows={4} />
  </Space>
);
```

## 错误与字数

```tsx preview
import { TextArea, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [value, setValue] = useState('这段内容可以一键清除');
  return (
    <Space direction="vertical" size="middle" className="w-full">
      <TextArea error="内容不能为空" placeholder="错误态" rows={3} />
      <TextArea
        value={value}
        onChange={setValue}
        showCount
        maxLength={50}
        clearable
        placeholder="showCount + maxLength"
        rows={3}
      />
    </Space>
  );
};
```
