# InputNumber 数字输入

## 基础用法

```tsx preview
import { InputNumber } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [v, setV] = useState(1);
  return <InputNumber value={v} onChange={setV} />;
};
```

## clearable

```tsx preview
import { InputNumber } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [v, setV] = useState(12);
  return <InputNumber value={v} onChange={setV} clearable />;
};
```

## 尺寸

```tsx preview
import { InputNumber, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <InputNumber size="sm" defaultValue={1} />
    <InputNumber size="md" defaultValue={1} />
    <InputNumber size="lg" defaultValue={1} />
  </Space>
);
```

## 最小 / 最大 / 步长

```tsx preview
import { InputNumber, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [v, setV] = useState(0);
  return (
    <div>
      <InputNumber value={v} onChange={setV} min={0} max={10} step={2} />
      <Text muted className="mt-2 block text-sm">当前：{v ?? '空'}（0–10，步长 2）</Text>
    </div>
  );
};
```

## 错误状态

```tsx preview
import { InputNumber, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <InputNumber error defaultValue={1} />
    <InputNumber error="超出允许范围" defaultValue={99} />
  </Space>
);
```

## 禁用

```tsx preview
import { InputNumber } from '@koi-ui/core';
export default () => <InputNumber disabled value={5} />;
```
