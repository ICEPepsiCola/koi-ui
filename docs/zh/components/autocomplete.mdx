# AutoComplete 自动完成

## 基础用法

```tsx preview
import { AutoComplete } from '@koi-ui/core';
export default () => <AutoComplete options={[{ value:'apple', label:'Apple' }, { value:'banana', label:'Banana' }]} placeholder="搜索" />;
```

## clearable

```tsx preview
import { AutoComplete } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [value, setValue] = useState('apple');
  return <AutoComplete value={value} onChange={setValue} clearable options={[{ value:'apple', label:'Apple' }, { value:'banana', label:'Banana' }]} placeholder="搜索" />;
};
```
