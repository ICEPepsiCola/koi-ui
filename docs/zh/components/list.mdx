
# List 列表

## 基础用法

```tsx preview
import { List } from '@koi-ui/core';
export default () => (
  <List
    items={[
      { key: '1', title: '标题一', description: '描述一' },
      { key: '2', title: '标题二', description: '描述二' },
    ]}
  />
);
```

## 边框与尺寸

```tsx preview
import { List, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <List
      bordered
      size="sm"
      items={[
        { key: '1', title: '小号列表', description: 'size=sm' },
        { key: '2', title: '另一项', description: '描述' },
      ]}
    />
    <List
      bordered
      size="lg"
      items={[
        { key: '1', title: '大号列表', description: 'size=lg' },
        { key: '2', title: '另一项', description: '描述' },
      ]}
    />
  </Space>
);
```
