
# TabBar 标签栏

## 基础用法

```tsx preview
import { TabBar } from '@koi-ui/core';
export default () => (
  <div className="relative h-16 overflow-hidden rounded-lg border border-border">
    <TabBar
      className="relative"
      defaultActiveKey="home"
      items={[
        { key: 'home', label: '首页' },
        { key: 'discover', label: '发现' },
        { key: 'mine', label: '我的' },
      ]}
    />
  </div>
);
```

## 受控

```tsx preview
import { TabBar, Icon, Space, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [activeKey, setActiveKey] = useState('home');
  return (
    <Space direction="vertical" size="middle" className="w-full">
      <Text size="sm" muted>activeKey: {activeKey}</Text>
      <div className="relative h-16 overflow-hidden rounded-lg border border-border">
        <TabBar
          className="relative"
          activeKey={activeKey}
          onChange={setActiveKey}
          items={[
            { key: 'home', label: '首页', icon: <Icon name="check" size="sm" /> },
            { key: 'search', label: '搜索', icon: <Icon name="search" size="sm" /> },
            { key: 'mine', label: '我的', icon: <Icon name="plus" size="sm" /> },
          ]}
        />
      </div>
    </Space>
  );
};
```

## 安全区

```tsx preview
import { TabBar, Icon, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <div className="relative h-16 overflow-hidden rounded-lg border border-border">
      <TabBar
        className="relative"
        safeArea
        defaultActiveKey="home"
        items={[
          { key: 'home', label: '开启', icon: <Icon name="check" size="sm" /> },
          { key: 'mine', label: '我的', icon: <Icon name="plus" size="sm" /> },
        ]}
      />
    </div>
    <div className="relative h-16 overflow-hidden rounded-lg border border-border">
      <TabBar
        className="relative"
        safeArea={false}
        defaultActiveKey="home"
        items={[
          { key: 'home', label: '关闭', icon: <Icon name="check" size="sm" /> },
          { key: 'mine', label: '我的', icon: <Icon name="plus" size="sm" /> },
        ]}
      />
    </div>
  </Space>
);
```
