# TabBar 标签栏

## 基础用法

```tsx preview
import { TabBar } from '@koi-ui/core';
export default () => (
  <TabBar
    defaultActiveKey="home"
    items={[
      { key: 'home', label: '首页' },
      { key: 'discover', label: '发现' },
      { key: 'mine', label: '我的' },
    ]}
  />
);
```

## 图标与徽标

```tsx preview
import { TabBar, Icon } from '@koi-ui/core';
export default () => (
  <TabBar
    defaultActiveKey="home"
    items={[
      { key: 'home', label: '首页', icon: <Icon name="check" size="sm" /> },
      { key: 'search', label: '搜索', icon: <Icon name="search" size="sm" />, badge: 3 },
      { key: 'mine', label: '我的', icon: <Icon name="plus" size="sm" />, badge: '新' },
    ]}
  />
);
```

## 禁用项

```tsx preview
import { TabBar, Icon } from '@koi-ui/core';
export default () => (

```

## 受控

```tsx preview
import { TabBar, Icon, Space, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [activeKey, setActiveKey] = useState('home');
  return (
    <Space direction="vertical" size="middle" className="w-full items-stretch">
      <Text size="sm" muted>activeKey: {activeKey}</Text>
      <TabBar
        activeKey={activeKey}
        onChange={setActiveKey}
        items={[
          { key: 'home', label: '首页', icon: <Icon name="check" size="sm" /> },
          { key: 'search', label: '搜索', icon: <Icon name="search" size="sm" /> },
          { key: 'mine', label: '我的', icon: <Icon name="plus" size="sm" /> },
        ]}
      />
    </Space>
  );
};
```

## 安全区

```tsx preview
import { TabBar, Icon, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full items-stretch">
    <TabBar
      safeArea
      defaultActiveKey="home"
      items={[
        { key: 'home', label: '开启', icon: <Icon name="check" size="sm" /> },
        { key: 'mine', label: '我的', icon: <Icon name="plus" size="sm" /> },
      ]}
    />
    <TabBar
      safeArea={false}
      defaultActiveKey="home"
      items={[
        { key: 'home', label: '关闭', icon: <Icon name="check" size="sm" /> },
        { key: 'mine', label: '我的', icon: <Icon name="plus" size="sm" /> },
      ]}
    />
  </Space>
);
```
