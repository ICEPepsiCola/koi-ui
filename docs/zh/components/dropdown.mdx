# Dropdown 下拉菜单

## 基础用法

```tsx preview
import { Dropdown, Button } from '@koi-ui/core';
export default () => (
  <Dropdown trigger={<Button>下拉菜单</Button>} items={[{ key: '1', label: '选项一' }, { key: '2', label: '选项二' }]} />
);
```

## 禁用项

```tsx preview
import { Dropdown, Button } from '@koi-ui/core';
export default () => (
  <Dropdown
    trigger={<Button>操作菜单</Button>}
    items={[
      { key: 'edit', label: '编辑' },
      { key: 'copy', label: '复制' },
      { key: 'delete', label: '删除', disabled: true },
    ]}
  />
);
```
