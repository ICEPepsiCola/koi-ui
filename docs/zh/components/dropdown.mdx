# Dropdown 下拉菜单

## 基础用法

```tsx preview
import { Dropdown, Button } from '@koi-ui/core';
export default () => (
  <Dropdown
    trigger={<Button>下拉菜单</Button>}
    items={[
      { key: '1', label: '选项一' },
      { key: '2', label: '选项二' },
    ]}
  />
);
```

## 禁用项

```tsx preview
import { Dropdown, Button } from '@koi-ui/core';
export default () => (
  <Dropdown
    trigger={<Button>操作菜单</Button>}
    items={[
      { key: 'edit', label: '编辑' },
      { key: 'copy', label: '复制' },
      { key: 'delete', label: '删除', disabled: true },
    ]}
  />
);
```

## 弹出位置

```tsx preview
import { Dropdown, Button, Space } from '@koi-ui/core';
const items = [
  { key: '1', label: '选项一' },
  { key: '2', label: '选项二' },
];
export default () => (
  <Space wrap>
    <Dropdown trigger={<Button>bottom-start</Button>} items={items} placement="bottom-start" />
    <Dropdown trigger={<Button>bottom-end</Button>} items={items} placement="bottom-end" />
    <Dropdown trigger={<Button>top-start</Button>} items={items} placement="top-start" />
    <Dropdown trigger={<Button>top-end</Button>} items={items} placement="top-end" />
  </Space>
);
```

## 标题（移动端）

```tsx preview
import { Dropdown, Button } from '@koi-ui/core';
export default () => (
  <Dropdown
    title="操作"
    trigger={<Button>带标题</Button>}
    items={[
      { key: 'edit', label: '编辑' },
      { key: 'share', label: '分享' },
    ]}
  />
);
```

## 禁用

```tsx preview
import { Dropdown, Button } from '@koi-ui/core';
export default () => (
  <Dropdown
    disabled
    trigger={<Button>已禁用</Button>}
    items={[
      { key: '1', label: '选项一' },
      { key: '2', label: '选项二' },
    ]}
  />
);
```
