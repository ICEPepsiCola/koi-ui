# Tabs 标签页

## 基础用法

```tsx preview
import { Tabs, Text } from '@koi-ui/core';
export default () => (
  <Tabs
    items={[
      { key: '1', label: 'Tab1', children: <Text>内容1</Text> },
      { key: '2', label: 'Tab2', children: <Text>内容2</Text> },
    ]}
  />
);
```

## 受控切换

```tsx preview
import { Tabs, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [key, setKey] = useState('profile');
  return (
    <Tabs
      activeKey={key}
      onChange={setKey}
      items={[
        { key: 'profile', label: '资料', children: <Text>个人资料面板</Text> },
        { key: 'settings', label: '设置', children: <Text>系统设置面板</Text> },
      ]}
    />
  );
};
```

## 默认激活

```tsx preview
import { Tabs, Text } from '@koi-ui/core';
export default () => (
  <Tabs
    defaultActiveKey="b"
    items={[
      { key: 'a', label: '标签 A', children: <Text>面板 A</Text> },
      { key: 'b', label: '标签 B', children: <Text>面板 B（默认）</Text> },
      { key: 'c', label: '标签 C', children: <Text>面板 C</Text> },
    ]}
  />
);
```

## 禁用项

```tsx preview
import { Tabs, Text } from '@koi-ui/core';
export default () => (
  <Tabs
    items={[
      { key: '1', label: '可用', children: <Text>可用面板</Text> },
      { key: '2', label: '禁用', children: <Text>不可见</Text>, disabled: true },
      { key: '3', label: '其他', children: <Text>其他面板</Text> },
    ]}
  />
);
```
