# Select 选择器

## 基础用法

```tsx preview
import { Select } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [v, setV] = useState('');
  return (
    <Select
      options={[
        { label: '北京', value: 'bj' },
        { label: '上海', value: 'sh' },
      ]}
      value={v}
      onChange={setV}
      placeholder="选择城市"
    />
  );
};
```

## 受控选择

```tsx preview
import { Select, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [city, setCity] = useState('sh');
  return (
    <div>
      <Select
        value={city}
        onChange={setCity}
        placeholder="选择城市"
        options={[
          { label: '北京', value: 'bj' },
          { label: '上海', value: 'sh' },
          { label: '广州', value: 'gz' },
        ]}
      />
      <Text muted className="mt-2 block text-sm">当前：{city}</Text>
    </div>
  );
};
```

## clearable

```tsx preview
import { Select } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [city, setCity] = useState('sh');
  return (
    <Select
      value={city}
      onChange={setCity}
      clearable

      options={[
        { label: '北京', value: 'bj' },
        { label: '上海', value: 'sh' },
        { label: '广州', value: 'gz' },
      ]}
    />
  );
};
```

## 尺寸

```tsx preview
import { Select, Space } from '@koi-ui/core';
const options = [
  { label: '北京', value: 'bj' },
  { label: '上海', value: 'sh' },
];
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <Select size="sm" options={options} placeholder="sm" />
    <Select size="md" options={options} placeholder="md" />
    <Select size="lg" options={options} placeholder="lg" />
  </Space>
);
```

## 禁用

```tsx preview
import { Select } from '@koi-ui/core';
export default () => (
  <Select
    disabled
    value="sh"
    options={[
      { label: '北京', value: 'bj' },
      { label: '上海', value: 'sh' },
    ]}
    placeholder="选择城市"
  />
);
```
