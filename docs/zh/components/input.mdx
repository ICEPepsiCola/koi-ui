# Input 输入框

## 基础用法

```tsx preview
import { Input } from '@koi-ui/core';
export default () => <Input placeholder="请输入" />;
```

## 错误提示

```tsx preview
import { Input, Stack } from '@koi-ui/core';
export default () => (
  <Stack gap={3}>
    <Input placeholder="正常" />
    <Input placeholder="出错" error="请输入有效内容" />
    <Input placeholder="禁用" disabled />
  </Stack>
);
```

## 受控输入

```tsx preview
import { Input } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [value, setValue] = useState('');
  return <Input value={value} onChange={setValue} placeholder="输入同步" />;
};
```

## 可清除

```tsx preview
import { Input } from '@koi-ui/core';
import { useState } from 'react';

export default () => {
  const [value, setValue] = useState('可清除内容');
  return (
    <Input
      value={value}
      onChange={setValue}
      clearable
      placeholder="可清除输入"
    />
  );
};
```

## 尺寸

```tsx preview
import { Input, Stack } from '@koi-ui/core';
export default () => (
  <Stack gap={3} className="max-w-md w-full">
    <Input size="sm" placeholder="Small (24px)" />
    <Input size="md" placeholder="Medium (32px)" />
    <Input size="lg" placeholder="Large (40px)" />
  </Stack>
);
```
