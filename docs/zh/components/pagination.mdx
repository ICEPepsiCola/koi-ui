# Pagination 分页

## 基础用法

```tsx preview
import { Pagination } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [page, setPage] = useState(1);
  return <Pagination current={page} total={50} onChange={setPage} />;
};
```

## 每页条数

```tsx preview
import { Pagination, Space, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [page, setPage] = useState(1);
  return (
    <Space direction="vertical" size="middle" className="w-full">
      <div>
        <Text muted size="sm" className="mb-2 block">pageSize=10 · total=100</Text>
        <Pagination current={page} total={100} pageSize={10} onChange={setPage} />
      </div>
      <div>
        <Text muted size="sm" className="mb-2 block">pageSize=20 · total=100</Text>
        <Pagination current={1} total={100} pageSize={20} />
      </div>
    </Space>
  );
};
```
