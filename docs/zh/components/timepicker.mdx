# TimePicker 时间选择

## 基础用法

```tsx preview
import { TimePicker } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [v,setV]=useState(''); return <TimePicker value={v} onChange={setV} placeholder="选择时间" />; };
```

## clearable

```tsx preview
import { TimePicker } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [v,setV]=useState('09:30'); return <TimePicker value={v} onChange={setV} clearable placeholder="选择时间" />; };
```
