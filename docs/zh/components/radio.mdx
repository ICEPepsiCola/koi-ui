# Radio 单选框

## 基础用法

```tsx preview
import { Radio, RadioGroup } from '@koi-ui/core';
export default () => (
  <RadioGroup defaultValue="a">
    <Radio radioValue="a" label="选项 A" />
    <Radio radioValue="b" label="选项 B" />
  </RadioGroup>
);
```

## 受控单选

```tsx preview
import { Radio, RadioGroup, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [value, setValue] = useState('a');
  return (
    <div>
      <RadioGroup value={value} onChange={setValue}>
        <Radio radioValue="a" label="方案 A" />
        <Radio radioValue="b" label="方案 B" />
      </RadioGroup>
      <Text muted className="mt-2 block text-sm">当前：{value}</Text>
    </div>
  );
};
```

## 尺寸、方向与禁用

```tsx preview
import { Radio, RadioGroup, Space, Text } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <div>
      <Text muted size="sm" className="mb-2 block">size</Text>
      <Space>
        <RadioGroup size="sm" defaultValue="sm" direction="horizontal">
          <Radio radioValue="sm" label="sm" />
        </RadioGroup>
        <RadioGroup size="md" defaultValue="md" direction="horizontal">
          <Radio radioValue="md" label="md" />
        </RadioGroup>
        <RadioGroup size="lg" defaultValue="lg" direction="horizontal">
          <Radio radioValue="lg" label="lg" />
        </RadioGroup>
      </Space>
    </div>
    <div>
      <Text muted size="sm" className="mb-2 block">direction=&quot;horizontal&quot;</Text>
      <RadioGroup defaultValue="a" direction="horizontal">
        <Radio radioValue="a" label="选项 A" />
        <Radio radioValue="b" label="选项 B" />
        <Radio radioValue="c" label="选项 C" />
      </RadioGroup>
    </div>
    <RadioGroup disabled defaultValue="a">
      <Radio radioValue="a" label="整组禁用" />
      <Radio radioValue="b" label="不可选" />
    </RadioGroup>
  </Space>
);
```
