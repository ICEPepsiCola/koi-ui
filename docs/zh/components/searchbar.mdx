## еҸ—жҺ§жҗңзҙў

```tsx preview
import { SearchBar, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [keyword, setKeyword] = useState('');
  return (
    <div>
      <SearchBar value={keyword} onChange={setKeyword} placeholder="жҗңзҙўе•Ҷе“Ғ" />
      <Text muted className="mt-2 block text-sm">е…ій”®иҜҚпјҡ{keyword || 'ж— '}</Text>
    </div>
  );
};
```

## clearable

```tsx preview
import { SearchBar } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [keyword, setKeyword] = useState('иҖіжңә');
  return <SearchBar value={keyword} onChange={setKeyword} clearable placeholder="жҗңзҙўе•Ҷе“Ғ" />;
};
```
