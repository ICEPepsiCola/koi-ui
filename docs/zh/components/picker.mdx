# Picker 选择器(移动)

## 基础用法

```tsx preview
import { Picker } from '@koi-ui/core';
export default () => (
  <Picker
    placeholder="选择"
    columns={[
      { options: [{ label: '周一', value: '1' }, { label: '周二', value: '2' }, { label: '周三', value: '3' }] },
      { options: [{ label: '上午', value: 'am' }, { label: '下午', value: 'pm' }] },
    ]}
  />
);
```

## clearable

```tsx preview
import { Picker } from '@koi-ui/core';
import { useState } from 'react';
const columns = [
  { options: [{ label: '周一', value: '1' }, { label: '周二', value: '2' }, { label: '周三', value: '3' }] },
  { options: [{ label: '上午', value: 'am' }, { label: '下午', value: 'pm' }] },
];
export default () => {
  const [value, setValue] = useState(['2', 'pm']);
  return <Picker value={value} onChange={setValue} columns={columns} clearable placeholder="选择时间段" />;
};
```

## 尺寸

```tsx preview
import { Picker, Space } from '@koi-ui/core';
const columns = [
  { options: [{ label: '周一', value: '1' }, { label: '周二', value: '2' }] },
  { options: [{ label: '上午', value: 'am' }, { label: '下午', value: 'pm' }] },
];
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <Picker size="sm" columns={columns} placeholder="sm" />
    <Picker size="md" columns={columns} placeholder="md" />
    <Picker size="lg" columns={columns} placeholder="lg" />
  </Space>
);
```

## 禁用

```tsx preview
import { Picker } from '@koi-ui/core';
export default () => (
  <Picker
    disabled
    value={['2', 'pm']}
    placeholder="已禁用"
    columns={[
      { options: [{ label: '周一', value: '1' }, { label: '周二', value: '2' }] },
      { options: [{ label: '上午', value: 'am' }, { label: '下午', value: 'pm' }] },
    ]}
  />
);
```

## 选项禁用

```tsx preview
import { Picker } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [value, setValue] = useState(['1', 'am']);
  return (
    <Picker
      value={value}
      onChange={setValue}
      placeholder="选择时间段"
      columns={[
        {
          options: [
            { label: '周一', value: '1' },
            { label: '周二', value: '2', disabled: true },
            { label: '周三', value: '3' },
          ],
        },
        {
          options: [
            { label: '上午', value: 'am' },
            { label: '下午', value: 'pm', disabled: true },
          ],
        },
      ]}
    />
  );
};
```
