# Toast 轻提示

## 基础用法

```tsx preview
import { Button, toast } from '@koi-ui/core';
export default () => (
  <Button onClick={() => toast('默认提示')}>显示提示</Button>
);
```

## 颜色

```tsx preview
import { Button, Space, toast } from '@koi-ui/core';
export default () => (
  <Space wrap>
    <Button onClick={() => toast('默认')}>default</Button>
    <Button variant="outline" onClick={() => toast.info('信息提示')}>info</Button>
    <Button variant="soft" onClick={() => toast.success('操作成功')}>success</Button>
    <Button color="secondary" onClick={() => toast.warning('请注意')}>warning</Button>
    <Button color="error" onClick={() => toast.error('出错了')}>error</Button>
  </Space>
);
```

## 位置

默认居中。可通过 `position` 指定顶部 / 底部。

```tsx preview
import { Button, Space, toast } from '@koi-ui/core';
export default () => (
  <Space wrap>
    <Button onClick={() => toast.show({ content: '居中', position: 'center' })}>
      居中
    </Button>
    <Button variant="outline" onClick={() => toast.show({ content: '顶部', position: 'top' })}>
      顶部
    </Button>
    <Button color="secondary" onClick={() => toast.show({ content: '底部', position: 'bottom' })}>
      底部
    </Button>
  </Space>
);
```

## 时长

```tsx preview
import { Button, Space, toast } from '@koi-ui/core';
export default () => (
  <Space wrap>
    <Button onClick={() => toast.success('2 秒后关闭', 2000)}>成功 2s</Button>
    <Button variant="outline" onClick={() => toast.info('自定义时长', 4000)}>信息 4s</Button>
    <Button
      color="secondary"
      onClick={() => toast.show({ content: '不自动关闭', duration: 0 })}
    >
      不自动关闭
    </Button>
  </Space>
);
```

## 清空

```tsx preview
import { Button, Space, toast } from '@koi-ui/core';
export default () => (
  <Space wrap>
    <Button
      onClick={() => {
        toast.info('第一条');
        toast.success('第二条', 5000);
        toast.warning('第三条', 5000);
      }}
    >
      连续弹出
    </Button>
    <Button variant="outline" onClick={() => toast.clear()}>
      清空全部
    </Button>
  </Space>
);
```
