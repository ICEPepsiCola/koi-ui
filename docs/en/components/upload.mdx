# Upload

## Basic usage

```tsx preview
import { Upload } from '@koi-ui/core';
export default () => <Upload accept="image/*" />;
```

## Custom hint

```tsx preview
import { Upload } from '@koi-ui/core';
export default () => (
  <Upload accept="image/*" hint="JPG/PNG only, max 5MB per file" />
);
```

## Multiple & maxCount

```tsx preview
import { Upload } from '@koi-ui/core';
export default () => (
  <Upload accept="image/*" multiple maxCount={3} hint="Up to 3 files" />
);
```

## Disabled

```tsx preview
import { Upload } from '@koi-ui/core';
export default () => <Upload disabled hint="Upload unavailable" />;
```
