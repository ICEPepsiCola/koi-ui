
# Rate

## Basic usage

```tsx preview
import { Rate } from '@koi-ui/core';
export default () => <Rate defaultValue={3} />;
```

## Half star

```tsx preview
import { Rate } from '@koi-ui/core';
export default () => <Rate allowHalf defaultValue={2.5} />;
```

## Count

```tsx preview
import { Rate } from '@koi-ui/core';
export default () => <Rate count={10} defaultValue={6} />;
```

## Colors

```tsx preview
import { Rate, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle">
    <Rate color="warning" defaultValue={4} />
    <Rate color="primary" defaultValue={3} />
    <Rate color="error" defaultValue={2} />
  </Space>
);
```

## Disabled

```tsx preview
import { Rate } from '@koi-ui/core';
export default () => <Rate disabled defaultValue={4} />;
```

## Controlled

```tsx preview
import { Rate, Space, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [value, setValue] = useState(3);
  return (
    <Space align="center">
      <Rate value={value} onChange={setValue} />
      <Text size="sm" muted>{value}</Text>
    </Space>
  );
};
```
