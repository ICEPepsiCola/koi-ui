
# Loading

## Basic usage

```tsx preview
import { Loading, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => {
          setOpen(true);
          setTimeout(() => setOpen(false), 1500);
        }}
      >
        Fullscreen loading
      </Button>
      <Loading open={open} tip="Loading..." />
    </>
  );
};
```

## Inline

```tsx preview
import { Loading } from '@koi-ui/core';
export default () => <Loading open fullscreen={false} tip="Please wait" />;
```

## Indicator styles

```tsx preview
import { Loading, Space } from '@koi-ui/core';
export default () => (
  <Space align="center" size="large">
    <Loading open fullscreen={false} indicator="spinner" tip="spinner" />
    <Loading open fullscreen={false} indicator="dots" tip="dots" />
    <Loading open fullscreen={false} indicator="ring" tip="ring" />
    <Loading open fullscreen={false} indicator="bars" tip="bars" />
  </Space>
);
```
