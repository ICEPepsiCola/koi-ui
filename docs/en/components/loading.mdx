# Loading

## Basic usage

```tsx preview
import { Loading, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [o,setO]=useState(false); return <><Button onClick={()=>setO(true)}>Load</Button><Loading open={o} /></>; };
```
