# Toast

## Basic usage

```tsx preview
import { Button, toast } from '@koi-ui/core';
export default () => (
  <Button onClick={() => toast('Default toast')}>Show toast</Button>
);
```

## Colors

```tsx preview
import { Button, Space, toast } from '@koi-ui/core';
export default () => (
  <Space wrap>
    <Button onClick={() => toast('Default')}>default</Button>
    <Button variant="outline" onClick={() => toast.info('Info toast')}>info</Button>
    <Button variant="soft" onClick={() => toast.success('Success')}>success</Button>
    <Button color="secondary" onClick={() => toast.warning('Warning')}>warning</Button>
    <Button color="error" onClick={() => toast.error('Error')}>error</Button>
  </Space>
);
```

## Position

Default is center. Pass `position` for top / bottom.

```tsx preview
import { Button, Space, toast } from '@koi-ui/core';
export default () => (
  <Space wrap>
    <Button onClick={() => toast.show({ content: 'Center', position: 'center' })}>
      Center
    </Button>
    <Button variant="outline" onClick={() => toast.show({ content: 'Top', position: 'top' })}>
      Top
    </Button>
    <Button color="secondary" onClick={() => toast.show({ content: 'Bottom', position: 'bottom' })}>
      Bottom
    </Button>
  </Space>
);
```

## Duration

```tsx preview
import { Button, Space, toast } from '@koi-ui/core';
export default () => (
  <Space wrap>
    <Button onClick={() => toast.success('Closes in 2s', 2000)}>Success 2s</Button>
    <Button variant="outline" onClick={() => toast.info('Custom duration', 4000)}>Info 4s</Button>
    <Button
      color="secondary"
      onClick={() => toast.show({ content: 'Does not auto-close', duration: 0 })}
    >
      No auto-close
    </Button>
  </Space>
);
```

## Clear

```tsx preview
import { Button, Space, toast } from '@koi-ui/core';
export default () => (
  <Space wrap>
    <Button
      onClick={() => {
        toast.info('First');
        toast.success('Second', 5000);
        toast.warning('Third', 5000);
      }}
    >
      Stack toasts
    </Button>
    <Button variant="outline" onClick={() => toast.clear()}>
      Clear all
    </Button>
  </Space>
);
```
