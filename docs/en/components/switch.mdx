
# Switch

## Basic usage

```tsx preview
import { Switch } from '@koi-ui/core';
export default () => <Switch defaultChecked />;
```

## Sizes

```tsx preview
import { Switch, Space } from '@koi-ui/core';
export default () => (
  <Space align="center">
    <Switch size="sm" defaultChecked />
    <Switch size="md" defaultChecked />
    <Switch size="lg" defaultChecked />
  </Space>
);
```

## Controlled

```tsx preview
import { Switch, Space, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [checked, setChecked] = useState(true);
  return (
    <Space align="center">
      <Switch checked={checked} onChange={setChecked} />
      <Text size="sm" muted>{checked ? 'On' : 'Off'}</Text>
    </Space>
  );
};
```

## Disabled

```tsx preview
import { Switch, Space } from '@koi-ui/core';
export default () => (
  <Space>
    <Switch disabled />
    <Switch disabled defaultChecked />
  </Space>
);
```

## Colors

```tsx preview
import { Switch, Space } from '@koi-ui/core';
export default () => (
  <Space wrap>
    <Switch color="primary" defaultChecked />
    <Switch color="secondary" defaultChecked />
    <Switch color="success" defaultChecked />
    <Switch color="warning" defaultChecked />
    <Switch color="error" defaultChecked />
  </Space>
);
```
