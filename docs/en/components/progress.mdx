
# Progress

## Basic usage

```tsx preview
import { Progress } from '@koi-ui/core';
export default () => <Progress percent={60} />;
```

## Colors

```tsx preview
import { Progress, Stack } from '@koi-ui/core';
export default () => (
  <Stack gap={4}>
    <Progress percent={30} showInfo />
    <Progress percent={70} color="success" showInfo />
    <Progress percent={55} color="warning" showInfo />
    <Progress percent={50} color="error" showInfo />
    <Progress percent={40} color="info" showInfo />
  </Stack>
);
```

## Size

```tsx preview
import { Progress, Stack } from '@koi-ui/core';
export default () => (
  <Stack gap={4}>
    <Progress percent={40} size="sm" showInfo />
    <Progress percent={60} size="md" showInfo />
    <Progress percent={80} size="lg" showInfo />
  </Stack>
);
```
