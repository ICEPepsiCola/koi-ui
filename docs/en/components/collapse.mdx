
# Collapse

## Basic usage

```tsx preview
import { Collapse } from '@koi-ui/core';
export default () => (
  <Collapse
    defaultActiveKeys={['1']}
    items={[
      { key: '1', label: 'Panel one', children: 'Content of panel one.' },
      { key: '2', label: 'Panel two', children: 'Content of panel two.' },
      { key: '3', label: 'Panel three', children: 'Content of panel three.' },
    ]}
  />
);
```

## Accordion

```tsx preview
import { Collapse } from '@koi-ui/core';
export default () => (
  <Collapse
    accordion
    defaultActiveKeys={['1']}
    items={[
      { key: '1', label: 'Only one open', children: 'With accordion, only one panel stays open.' },
      { key: '2', label: 'Another', children: 'Opening this closes the previous one.' },
      { key: '3', label: 'Disabled', children: 'Cannot expand', disabled: true },
    ]}
  />
);
```

## Controlled

```tsx preview
import { Collapse, Space, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [activeKeys, setActiveKeys] = useState(['1']);
  return (
    <Space direction="vertical" size="middle" className="w-full">
      <Text size="sm" muted>activeKeys: {activeKeys.join(', ') || '(empty)'}</Text>
      <Collapse
        activeKeys={activeKeys}
        onChange={setActiveKeys}
        items={[
          { key: '1', label: 'Controlled A', children: 'Driven by activeKeys / onChange.' },
          { key: '2', label: 'Controlled B', children: 'Multiple panels can stay open.' },
        ]}
      />
    </Space>
  );
};
```
