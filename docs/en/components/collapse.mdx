# Collapse

## Basic usage

```tsx preview
import { Collapse } from '@koi-ui/core';
export default () => <Collapse items={[{ key:'1', label:'Panel 1', children:'Content 1' }, { key:'2', label:'Panel 2', children:'Content 2' }]} />;
```
