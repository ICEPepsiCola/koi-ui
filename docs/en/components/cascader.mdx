## Size

```tsx preview
import { Cascader, Space } from '@koi-ui/core';
const options = [
  {
    label: 'Zhejiang',
    value: 'zj',
    children: [
      {
        label: 'Hangzhou',
        value: 'hz',
        children: [
          { label: 'Xihu', value: 'xh' },
          { label: 'Yuhang', value: 'yh' },
        ],
      },
    ],
  },
  {
    label: 'Jiangsu',
    value: 'js',
    children: [
      {
        label: 'Nanjing',
        value: 'nj',
        children: [
          { label: 'Xuanwu', value: 'xw' },
          { label: 'Qinhuai', value: 'qh' },
        ],
      },
    ],
  },
];
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <Cascader size="sm" options={options} placeholder="sm" />
    <Cascader size="md" options={options} placeholder="md" />
    <Cascader size="lg" options={options} placeholder="lg" />
  </Space>
);
```
