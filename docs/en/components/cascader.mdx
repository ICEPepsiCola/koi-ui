# Cascader

## Basic usage

```tsx preview
import { Cascader } from '@koi-ui/core';
export default () => <Cascader options={[{label:'Zhejiang',value:'zj',children:[{label:'Hangzhou',value:'hz'}]}]} placeholder="Select region" />;
```

## clearable

```tsx preview
import { Cascader } from '@koi-ui/core';
import { useState } from 'react';
const options = [{ label:'Zhejiang', value:'zj', children:[{ label:'Hangzhou', value:'hz' }] }];
export default () => {
  const [value, setValue] = useState(['zj', 'hz']);
  return <Cascader value={value} onChange={setValue} options={options} clearable placeholder="Select region" />;
};
```
