# Modal

Desktop: centered dialog (daisyUI `modal-middle`). Mobile: bottom sheet when `responsive`.

## Basic usage

```tsx preview
import { Modal, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>Open modal</Button>
      <Modal open={open} onClose={() => setOpen(false)} title="Hello!">
        Press ESC or click the mask to close.
      </Modal>
    </>
  );
};
```

## Actions & close button

```tsx preview
import { Modal, Button, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>Confirm</Button>
      <Modal
        open={open}
        onClose={() => setOpen(false)}
        title="Confirm delete"
        closable
        footer={
          <Space>
            <Button color="neutral" variant="ghost" onClick={() => setOpen(false)}>
              Cancel
            </Button>
            <Button color="error" onClick={() => setOpen(false)}>
              Delete
            </Button>
          </Space>
        }
      >
        Delete this record?
      </Modal>
    </>
  );
};
```

## Placement

Desktop placements mirror daisyUI. Force desktop layout with `responsive={false}`.

```tsx preview
import { Modal, Button, Space } from '@koi-ui/core';
import { useState } from 'react';

type Placement = 'middle' | 'top' | 'bottom' | 'start' | 'end';

export default () => {
  const [placement, setPlacement] = useState<Placement>('middle');
  const [open, setOpen] = useState(false);
  return (
    <>
      <Space wrap>
        {(['middle', 'top', 'bottom', 'start', 'end'] as const).map((p) => (
          <Button
            key={p}
            size="sm"
            variant={placement === p ? 'solid' : 'outline'}
            onClick={() => {
              setPlacement(p);
              setOpen(true);
            }}
          >
            {p}
          </Button>
        ))}
      </Space>
      <Modal
        open={open}
        onClose={() => setOpen(false)}
        title={`placement="${placement}"`}
        placement={placement}
        responsive={false}
        closable
      >
        Desktop placement demo (responsive off).
      </Modal>
    </>
  );
};
```

## Mobile height

```tsx preview
import { Modal, Button, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [tall, setTall] = useState(false);
  const [open, setOpen] = useState(false);
  return (
    <>
      <Space>
        <Button
          onClick={() => {
            setTall(true);
            setOpen(true);
          }}
        >
          Near fullscreen
        </Button>
        <Button
          variant="outline"
          onClick={() => {
            setTall(false);
            setOpen(true);
          }}
        >
          Shorter
        </Button>
      </Space>
      <Modal
        open={open}
        onClose={() => setOpen(false)}
        title="Mobile drawer height"
        mobileFullscreen={tall}
      >
        On mobile, Modal renders as a bottom sheet. mobileFullscreen true ≈
        90vh; false ≈ 70vh.
      </Modal>
    </>
  );
};
```
