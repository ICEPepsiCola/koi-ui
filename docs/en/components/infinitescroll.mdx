# InfiniteScroll

## Basic usage

```tsx preview
import { InfiniteScroll, List } from '@koi-ui/core';
export default () => <InfiniteScroll hasMore={false} loadMore={async()=>{}}><List items={[{key:'1',title:'Item 1'},{key:'2',title:'Item 2'}]} /></InfiniteScroll>;
```
