# InfiniteScroll

## Basic usage

```tsx preview
import { InfiniteScroll, List, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [items, setItems] = useState(
    Array.from({ length: 8 }, (_, i) => ({ key: String(i), title: `Item ${i + 1}` })),
  );
  const [hasMore, setHasMore] = useState(true);
  return (
    <div className="space-y-2">
      <Text muted size="sm">Scroll to the bottom to load more</Text>
      <InfiniteScroll
        className="h-64 rounded-md border border-border"
        hasMore={hasMore}
        loadMore={async () => {
          await new Promise((r) => setTimeout(r, 600));
          setItems((prev) => {
            const next = [
              ...prev,
              ...Array.from({ length: 5 }, (_, i) => ({
                key: String(prev.length + i),
                title: `Item ${prev.length + i + 1}`,
              })),
            ];
            if (next.length >= 25) setHasMore(false);
            return next;
          });
        }}
      >
        <List items={items} />
      </InfiniteScroll>
    </div>
  );
};
```

## loadingText & endText

```tsx preview
import { InfiniteScroll, List } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [items, setItems] = useState(
    Array.from({ length: 6 }, (_, i) => ({ key: String(i), title: `Row ${i + 1}` })),
  );
  const [hasMore, setHasMore] = useState(true);
  return (
    <InfiniteScroll
      className="h-64 rounded-md border border-border"
      hasMore={hasMore}
      loadingText="Loading more…"
      endText="That's all"
      loadMore={async () => {
        await new Promise((r) => setTimeout(r, 500));
        setItems((prev) => {
          const next = [
            ...prev,
            ...Array.from({ length: 4 }, (_, i) => ({
              key: String(prev.length + i),
              title: `Row ${prev.length + i + 1}`,
            })),
          ];
          if (next.length >= 18) setHasMore(false);
          return next;
        });
      }}
    >
      <List items={items} />
    </InfiniteScroll>
  );
};
```
