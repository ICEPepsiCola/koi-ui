# Menu

## Basic usage

```tsx preview
import { Menu } from '@koi-ui/core';
export default () => <Menu items={[{ key: '1', label: 'Nav 1' }, { key: '2', label: 'Nav 2' }]} defaultSelectedKey="1" />;
```
