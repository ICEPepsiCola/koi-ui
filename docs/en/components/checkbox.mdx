# Checkbox

## Basic usage

```tsx preview
import { Checkbox, CheckboxGroup } from '@koi-ui/core';
export default () => (
  <CheckboxGroup defaultValue={['a']}>
    <Checkbox checkboxValue="a" label="Option A" />
    <Checkbox checkboxValue="b" label="Option B" />
  </CheckboxGroup>
);
```

## Controlled checkbox group

```tsx preview
import { Checkbox, CheckboxGroup, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [values, setValues] = useState(['read']);
  return (
    <div>
      <CheckboxGroup value={values} onChange={setValues}>
        <Checkbox checkboxValue="read" label="Read" />
        <Checkbox checkboxValue="write" label="Write" />
        <Checkbox checkboxValue="code" label="Code" />
      </CheckboxGroup>
      <Text muted className="mt-2 block text-sm">Selected: {values.join(', ') || 'None'}</Text>
    </div>
  );
};
```

## Size & disabled

```tsx preview
import { Checkbox, CheckboxGroup, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <CheckboxGroup size="sm" defaultValue={['sm']}>
      <Checkbox checkboxValue="sm" label="sm" />
      <Checkbox checkboxValue="b" label="Option" />
    </CheckboxGroup>
    <CheckboxGroup size="md" defaultValue={['md']}>
      <Checkbox checkboxValue="md" label="md" />
      <Checkbox checkboxValue="b" label="Option" />
    </CheckboxGroup>
    <CheckboxGroup size="lg" defaultValue={['lg']}>
      <Checkbox checkboxValue="lg" label="lg" />
      <Checkbox checkboxValue="b" label="Option" />
    </CheckboxGroup>
    <CheckboxGroup disabled defaultValue={['a']}>
      <Checkbox checkboxValue="a" label="Group disabled" />
      <Checkbox checkboxValue="b" label="Unavailable" />
    </CheckboxGroup>
    <Checkbox label="Item disabled" disabled defaultChecked />
  </Space>
);
```
