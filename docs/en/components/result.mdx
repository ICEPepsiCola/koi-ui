# Result

## Basic usage

```tsx preview
import { Result, Button } from '@koi-ui/core';
export default () => <Result status="success" title="Submitted" extra={<Button>Back</Button>} />;
```

## Error result

```tsx preview
import { Result, Button } from '@koi-ui/core';
export default () => (
  <Result
    status="error"
    title="Submit failed"
    subTitle="Check your network and retry"
    extra={<Button variant="primary">Retry</Button>}
  />
);
```
