# Tabs

## Basic usage

```tsx preview
import { Tabs, Text } from '@koi-ui/core';
export default () => (
  <Tabs
    items={[
      { key: '1', label: 'Tab1', children: <Text>Content 1</Text> },
      { key: '2', label: 'Tab2', children: <Text>Content 2</Text> },
    ]}
  />
);
```

## Controlled tabs

```tsx preview
import { Tabs, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [key, setKey] = useState('profile');
  return (
    <Tabs
      activeKey={key}
      onChange={setKey}
      items={[
        { key: 'profile', label: 'Profile', children: <Text>Profile panel</Text> },
        { key: 'settings', label: 'Settings', children: <Text>Settings panel</Text> },
      ]}
    />
  );
};
```

## Default active

```tsx preview
import { Tabs, Text } from '@koi-ui/core';
export default () => (
  <Tabs
    defaultActiveKey="b"
    items={[
      { key: 'a', label: 'Tab A', children: <Text>Panel A</Text> },
      { key: 'b', label: 'Tab B', children: <Text>Panel B (default)</Text> },
      { key: 'c', label: 'Tab C', children: <Text>Panel C</Text> },
    ]}
  />
);
```

## Disabled items

```tsx preview
import { Tabs, Text } from '@koi-ui/core';
export default () => (
  <Tabs
    items={[
      { key: '1', label: 'Enabled', children: <Text>Enabled panel</Text> },
      { key: '2', label: 'Disabled', children: <Text>Hidden</Text>, disabled: true },
      { key: '3', label: 'Other', children: <Text>Other panel</Text> },
    ]}
  />
);
```

## Variants

```tsx preview
import { Tabs, Text, Space } from '@koi-ui/core';
const items = [
  { key: '1', label: 'Tab1', children: <Text>Panel 1</Text> },
  { key: '2', label: 'Tab2', children: <Text>Panel 2</Text> },
  { key: '3', label: 'Tab3', children: <Text>Panel 3</Text> },
];
export default () => (
  <Space direction="vertical" size="large" className="w-full">
    <Tabs variant="line" items={items} />
    <Tabs variant="boxed" items={items} />
    <Tabs variant="bordered" items={items} />
    <Tabs variant="lifted" items={items} />
  </Space>
);
```
