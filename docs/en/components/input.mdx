
# Input

## Basic usage

```tsx preview
import { Input } from '@koi-ui/core';
export default () => <Input placeholder="Enter text" />;
```

## Error state

```tsx preview
import { Input, Stack } from '@koi-ui/core';
export default () => (
  <Stack gap={3}>
    <Input placeholder="Normal" />
    <Input placeholder="Invalid" error="Enter valid content" />
    <Input placeholder="Disabled" disabled />
  </Stack>
);
```

## Controlled input

```tsx preview
import { Input } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [value, setValue] = useState('');
  return <Input value={value} onChange={setValue} placeholder="Synced input" />;
};
```

## Clearable

```tsx preview
import { Input } from '@koi-ui/core';
import { useState } from 'react';

export default () => {
  const [value, setValue] = useState('Clearable text');
  return (
    <Input
      value={value}
      onChange={setValue}
      clearable
      placeholder="Clearable input"
    />
  );
};
```

## Sizes

```tsx preview
import { Input, Stack } from '@koi-ui/core';
export default () => (
  <Stack gap={3} className="max-w-md w-full">
    <Input size="sm" placeholder="Small (24px)" />
    <Input size="md" placeholder="Medium (32px)" />
    <Input size="lg" placeholder="Large (40px)" />
  </Stack>
);
```
