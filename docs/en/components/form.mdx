
# Form

## Basic usage

```tsx preview
import { Form, Input, Button, toast } from '@koi-ui/core';

export default () => (
  <Form
    layout="vertical"
    onFinish={(values) => toast.success(`Hello ${values.username}`)}
  >
    <Form.Item
      name="username"
      label="Username"
      rules={[{ required: true, message: 'Enter textUsername' }]}
    >
      <Input placeholder="Enter text" clearable />
    </Form.Item>
    <Button color="primary" type="submit">Submit</Button>
  </Form>
);
```

## Rules & form instance

```tsx preview
import { Form, Input, InputNumber, Switch, Button, Space, toast } from '@koi-ui/core';

export default () => {
  const [form] = Form.useForm();
  return (
    <Form
      form={form}
      layout="vertical"
      initialValues={{ age: 18, notify: true }}
      onFinish={(values) => toast.success(JSON.stringify(values))}
      onFinishFailed={() => toast.error('请检查表单')}
    >
      <Form.Item
        name="email"
        label="Email"
        rules={[
          { required: true, message: '必填' },
          { type: 'email', message: 'Email格式不正确' },
        ]}
      >
        <Input placeholder="name@example.com" clearable />
      </Form.Item>
      <Form.Item name="age" label="Age" rules={[{ type: 'number', min: 1, max: 120, message: '1-120' }]}>
        <InputNumber />
      </Form.Item>
      <Form.Item name="notify" label="Notice" valuePropName="checked">
        <Switch />
      </Form.Item>
      <Space>
        <Button color="primary" type="submit">Submit</Button>
        <Button type="button" variant="outline" onClick={() => form.resetFields()}>重置</Button>
      </Space>
    </Form>
  );
};
```

## Horizontal layout

```tsx preview
import { Form, Input, Button, Space } from '@koi-ui/core';

export default () => (
  <Form
    layout="horizontal"
    onFinish={() => {}}
  >
    <Form.Item name="email" label="Email" rules={[{ required: true, message: '必填' }]}>
      <Input placeholder="name@example.com" />
    </Form.Item>
    <Form.Item name="password" label="Password" rules={[{ required: true, min: 6, message: '至少 6 位' }]}>
      <Input type="password" placeholder="Enter password" />
    </Form.Item>
    <Space>
      <Button color="primary" type="submit">Sign in</Button>
      <Button type="button" color="secondary">Sign up</Button>
    </Space>
  </Form>
);
```

## Unified field sizes

```tsx preview
import {
  AutoComplete,
  Cascader,
  Input,
  SearchBar,
  Select,
  Stack,
  Text,
} from '@koi-ui/core';

const regionOptions = [
  { label: 'Zhejiang', value: 'zj', children: [{ label: 'Hangzhou', value: 'hz' }] },
];

export default () => (
  <Stack gap={3} className="max-w-md w-full">
    <Text muted size="sm">Default md — 单行表单控件统一 32px 高度（同 Ant Design medium）</Text>
    <Input placeholder="输入框" />
    <Select
      placeholder="Select city"
      options={[
        { label: 'Beijing', value: 'bj' },
        { label: 'Shanghai', value: 'sh' },
      ]}
    />
    <Cascader options={regionOptions} placeholder="Select region" />
    <SearchBar placeholder="Search" />
    <AutoComplete
      placeholder="自动Done"
      options={[
        { value: 'apple', label: 'Apple' },
        { value: 'banana', label: 'Banana' },
      ]}
    />
    <Text muted size="sm" className="pt-2">size="sm" — 24px</Text>
    <Input size="sm" placeholder="输入框" />
    <Select
      size="sm"
      placeholder="Select city"
      options={[{ label: 'Beijing', value: 'bj' }, { label: 'Shanghai', value: 'sh' }]}
    />
    <Cascader size="sm" options={regionOptions} placeholder="Select region" />
    <SearchBar size="sm" placeholder="Search" />
  </Stack>
);
```
