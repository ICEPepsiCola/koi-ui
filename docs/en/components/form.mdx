# Form

## Basic usage

```tsx preview
import { Form, FormItem, Input, Button } from '@koi-ui/core';
export default () => (
  <Form layout="vertical"><FormItem label="Username"><Input placeholder="Enter text" /></FormItem><Button variant="primary">Submit</Button></Form>
);
```

## Horizontal layout

```tsx preview
import { Form, FormItem, Input, Button, Space } from '@koi-ui/core';
export default () => (
  <Form layout="horizontal">
    <FormItem label="Email"><Input placeholder="name@example.com" /></FormItem>
    <FormItem label="Password"><Input type="password" placeholder="Enter password" /></FormItem>
    <Space><Button variant="primary">Sign in</Button><Button variant="secondary">Sign up</Button></Space>
  </Form>
);
```
