# Badge

## Basic usage

```tsx preview
import { Badge, Button, Space } from '@koi-ui/core';
export default () => (
  <Space>
    <Badge count={5}><Button>Messages</Button></Badge>
    <Badge count={99}><Button>Alerts</Button></Badge>
    <Badge count={100} max={99}><Button>Overflow</Button></Badge>
  </Space>
);
```

## Colors & variants

```tsx preview
import { Badge, Button, Space } from '@koi-ui/core';
export default () => (
  <Space wrap>
    <Badge count={5} color="error"><Button>error</Button></Badge>
    <Badge count={3} color="primary" variant="soft"><Button variant="soft">soft</Button></Badge>
    <Badge count={9} color="secondary"><Button color="secondary">secondary</Button></Badge>
    <Badge count={4} color="info" variant="outline"><Button variant="outline">outline</Button></Badge>
    <Badge count={1} color="success" variant="ghost"><Button variant="ghost">ghost</Button></Badge>
  </Space>
);
```

## Sizes & dot

```tsx preview
import { Badge, Button, Space } from '@koi-ui/core';
export default () => (
  <Space wrap>
    <Badge count={5} size="sm"><Button size="sm">sm</Button></Badge>
    <Badge count={5} size="md"><Button>md</Button></Badge>
    <Badge count={5} size="lg"><Button size="lg">lg</Button></Badge>
    <Badge dot><Button>Dot</Button></Badge>
    <Badge count={0} showZero><Button>Zero</Button></Badge>
  </Space>
);
```

## Standalone

```tsx preview
import { Badge, Space } from '@koi-ui/core';
export default () => (
  <Space>
    <Badge count={8} />
    <Badge count={120} max={99} />
    <Badge dot />
  </Space>
);
```
