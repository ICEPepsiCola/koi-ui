# DatePicker

## Basic usage

```tsx preview
import { DatePicker } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [v, setV] = useState('');
  return <DatePicker value={v} onChange={setV} placeholder="Select date" />;
};
```

## Controlled date

```tsx preview
import { DatePicker, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [date, setDate] = useState('2026-07-12');
  return (
    <div>
      <DatePicker value={date} onChange={setDate} placeholder="Select date" />
      <Text muted className="mt-2 block text-sm">Selected: {date || 'None'}</Text>
    </div>
  );
};
```

## clearable

```tsx preview
import { DatePicker } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [v, setV] = useState('2026-07-12');
  return <DatePicker value={v} onChange={setV} clearable placeholder="Select date" />;
};
```

## Min / max

```tsx preview
import { DatePicker } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [v, setV] = useState('2026-07-15');
  return (
    <DatePicker
      value={v}
      onChange={setV}
      min="2026-07-01"
      max="2026-07-31"
      placeholder="July only"
    />
  );
};
```

## Size

```tsx preview
import { DatePicker, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <DatePicker size="sm" placeholder="sm" />
    <DatePicker size="md" placeholder="md" />
    <DatePicker size="lg" placeholder="lg" />
  </Space>
);
```

## Disabled

```tsx preview
import { DatePicker } from '@koi-ui/core';
export default () => (
  <DatePicker disabled value="2026-07-12" placeholder="Select date" />
);
```
