# BackTop

## Basic usage

```tsx preview
import { BackTop, Text } from '@koi-ui/core';
export default () => (
  <div className="relative space-y-2 pb-8 text-sm text-muted-foreground">
    <Text className="font-medium text-surface-foreground">Scroll down to reveal BackTop</Text>
    {Array.from({ length: 50 }, (_, i) => (
      <p key={i}>Row {i + 1}</p>
    ))}
    <BackTop visibilityHeight={120} />
  </div>
);
```

## visibilityHeight

```tsx preview
import { BackTop, Text } from '@koi-ui/core';
export default () => (
  <div className="relative space-y-2 pb-8 text-sm text-muted-foreground">

    {Array.from({ length: 50 }, (_, i) => (
      <p key={i}>Row {i + 1}</p>
    ))}
    <BackTop visibilityHeight={40} />
  </div>
);
```
