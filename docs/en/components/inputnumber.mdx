
# InputNumber

## Basic usage

```tsx preview
import { InputNumber } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [v,setV]=useState(1); return <InputNumber value={v} onChange={setV} />; };
```

## clearable

```tsx preview
import { InputNumber } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [v,setV]=useState(12); return <InputNumber value={v} onChange={setV} clearable />; };
```
