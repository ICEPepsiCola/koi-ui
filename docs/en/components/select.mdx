# Select

## Basic usage

```tsx preview
import { Select } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [v,setV]=useState(''); return <Select options={[{label:'Beijing',value:'bj'},{label:'Shanghai',value:'sh'}]} value={v} onChange={setV} placeholder="Select city" />; };
```

## Controlled select

```tsx preview
import { Select, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [city, setCity] = useState('sh');
  return (
    <div>
      <Select
        value={city}
        onChange={setCity}
        placeholder="Select city"
        options={[
          { label: 'Beijing', value: 'bj' },
          { label: 'Shanghai', value: 'sh' },
          { label: 'Guangzhou', value: 'gz' },
        ]}
      />
      <Text muted className="mt-2 block text-sm">Current: {city}</Text>
    </div>
  );
};
```
