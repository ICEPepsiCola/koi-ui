# Popup

## Basic usage

```tsx preview
import { Popup, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [o,setO]=useState(false); return <><Button onClick={()=>setO(true)}>Popup</Button><Popup open={o} onClose={()=>setO(false)}>Popup content</Popup></>; };
```
