# Popup

Compact centered dialog — same chrome as Modal, default `size="sm"`.

## Basic usage

```tsx preview
import { Popup, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>Open popup</Button>
      <Popup open={open} onClose={() => setOpen(false)} title="Notice">
        Centered popup content.
      </Popup>
    </>
  );
};
```

## Actions & close button

```tsx preview
import { Popup, Button, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>With footer</Button>
      <Popup
        open={open}
        onClose={() => setOpen(false)}
        title="Confirm"
        closable
        footer={
          <Space>
            <Button color="neutral" variant="ghost" size="sm" onClick={() => setOpen(false)}>
              Cancel
            </Button>
            <Button color="primary" size="sm" onClick={() => setOpen(false)}>
              OK
            </Button>
          </Space>
        }
      >
        Continue?
      </Popup>
    </>
  );
};
```

## Mask does not close

```tsx preview
import { Popup, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>Mask locked</Button>
      <Popup
        open={open}
        onClose={() => setOpen(false)}
        title="Important"
        maskClosable={false}
        closable
        footer={
          <Button size="sm" onClick={() => setOpen(false)}>
            Got it
          </Button>
        }
      >
        When maskClosable is false, close via the button.
      </Popup>
    </>
  );
};
```
