
# TimePicker

## Basic usage

```tsx preview
import { TimePicker } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [v,setV]=useState(''); return <TimePicker value={v} onChange={setV} placeholder="Select time" />; };
```

## clearable

```tsx preview
import { TimePicker } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [v,setV]=useState('09:30'); return <TimePicker value={v} onChange={setV} clearable placeholder="Select time" />; };
```
