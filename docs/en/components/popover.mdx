# Popover

## Basic usage

```tsx preview
import { Popover, Button } from '@koi-ui/core';
export default () => (
  <div className="flex justify-center py-8">
    <Popover content="Popover content">
      <Button>Click</Button>
    </Popover>
  </div>
);
```

## Title

```tsx preview
import { Popover, Button } from '@koi-ui/core';
export default () => (
  <div className="flex justify-center py-8">
    <Popover title="Title" content="Popover body with a title.">
      <Button variant="outline">With title</Button>
    </Popover>
  </div>
);
```

## Placement

```tsx preview
import { Popover, Button } from '@koi-ui/core';

const cell = 'flex h-24 items-center justify-center';

export default () => (
  <div className="grid grid-cols-2 gap-4 sm:grid-cols-4">
    <div className={cell}>
      <Popover placement="top" content="Top" defaultOpen>
        <Button size="sm">top</Button>
      </Popover>
    </div>
    <div className={cell}>
      <Popover placement="bottom" content="Bottom" defaultOpen>
        <Button size="sm">bottom</Button>
      </Popover>
    </div>
    <div className={cell}>
      <Popover placement="left" content="Left" defaultOpen>
        <Button size="sm">left</Button>
      </Popover>
    </div>
    <div className={cell}>
      <Popover placement="right" content="Right" defaultOpen>
        <Button size="sm">right</Button>
      </Popover>
    </div>
  </div>
);
```

## Hover trigger

```tsx preview
import { Popover, Button } from '@koi-ui/core';
export default () => (
  <div className="flex justify-center py-8">
    <Popover trigger="hover" content="Shown on hover">
      <Button variant="soft">Hover me</Button>
    </Popover>
  </div>
);
```
