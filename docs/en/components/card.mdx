
# Card

## Basic usage

```tsx preview
import { Card, Text } from '@koi-ui/core';
export default () => (
  <Card title="Card title">
    <Text>Card content</Text>
  </Card>
);
```

## Padding

```tsx preview
import { Card, Text, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <Card padding="sm" title="sm"><Text size="sm">Small padding</Text></Card>
    <Card padding="md" title="md"><Text size="sm">Default padding</Text></Card>
    <Card padding="lg" title="lg"><Text size="sm">Large padding</Text></Card>
  </Space>
);
```

## Cover & actions

```tsx preview
import { Card, Button, Text, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <Card
      cover={
        <img
          src="https://picsum.photos/seed/koi/640/200"
          alt="cover"
          className="h-32 w-full object-cover"
        />
      }
      title="With cover"
      actions={
        <>
          <Button size="sm" variant="ghost">Share</Button>
          <Button size="sm" color="primary">Details</Button>
        </>
      }
    >
      <Text>cover and actions structure.</Text>
    </Card>
    <Card
      title="Project overview"
      footer={<Button color="primary" size="sm">View details</Button>}
    >
      <Text>Card with a footer action area.</Text>
    </Card>
    <Card title="Hoverable" hoverable onClick={() => {}}>
      <Text>hoverable adds elevation feedback.</Text>
    </Card>
  </Space>
);
```
