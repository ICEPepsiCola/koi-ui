# NavBar

## Basic usage

```tsx preview
import { NavBar } from '@koi-ui/core';
export default () => <NavBar title="Page title" onBack={() => {}} />;
```

## With right action

```tsx preview
import { NavBar, Button } from '@koi-ui/core';
export default () => (
  <NavBar
    title="Details"
    onBack={() => {}}
    right={<Button size="sm" variant="ghost">Share</Button>}
  />
);
```
