# Drawer

Side / edge panel with the same chrome as Modal (no section borders, optional `closable`).

## Basic usage

```tsx preview
import { Drawer, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>Open drawer</Button>
      <Drawer open={open} onClose={() => setOpen(false)} title="Drawer" closable>
        Content
      </Drawer>
    </>
  );
};
```

## Placement

```tsx preview
import { Drawer, Button, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [placement, setPlacement] = useState<'left' | 'right' | 'top' | 'bottom'>('right');
  const [open, setOpen] = useState(false);
  return (
    <>
      <Space wrap>
        {(['left', 'right', 'top', 'bottom'] as const).map((p) => (
          <Button
            key={p}
            size="sm"
            variant={placement === p ? 'solid' : 'outline'}
            onClick={() => {
              setPlacement(p);
              setOpen(true);
            }}
          >
            {p}
          </Button>
        ))}
      </Space>
      <Drawer
        open={open}
        onClose={() => setOpen(false)}
        title={`placement=${placement}`}
        placement={placement}
        closable
      >
        Slides in from {placement}
      </Drawer>
    </>
  );
};
```

```tsx preview
};
```

## Footer actions

```tsx preview
import { Drawer, Button, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>With footer</Button>
      <Drawer
        open={open}
        onClose={() => setOpen(false)}
        title="Edit"
        closable
        footer={
          <Space>
            <Button color="neutral" variant="ghost" onClick={() => setOpen(false)}>
              Cancel
            </Button>
            <Button color="primary" onClick={() => setOpen(false)}>
              Save
            </Button>
          </Space>
        }
      >
        Drawer body with footer actions
      </Drawer>
    </>
  );
};
```
