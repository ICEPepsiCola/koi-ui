# Mask

## Basic usage

```tsx preview
import { Mask, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [o,setO]=useState(false); return <><Button onClick={()=>setO(true)}>Show mask</Button><Mask open={o} onClick={()=>setO(false)} /></>; };
```
