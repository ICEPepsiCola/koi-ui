# Mask

## Basic usage

```tsx preview
import { Mask, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>Show mask</Button>
      <Mask open={open} onClick={() => setOpen(false)} />
    </>
  );
};
```

## zIndex & children

```tsx preview
import { Mask, Button, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>Mask with content</Button>
      <Mask open={open} zIndex={80} onClick={() => setOpen(false)}>
        <div
          className="absolute left-1/2 top-1/2 w-64 -translate-x-1/2 -translate-y-1/2 rounded-box bg-surface p-4 shadow-float"
          onClick={(e) => e.stopPropagation()}
        >
          <Text className="mb-3 block font-medium">Overlay content</Text>
          <Text muted size="sm" className="mb-4 block">
            Click the mask to close, or use the button below.
          </Text>
          <Button size="sm" onClick={() => setOpen(false)}>Close</Button>
        </div>
      </Mask>
    </>
  );
};
```
