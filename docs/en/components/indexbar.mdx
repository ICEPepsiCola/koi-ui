# IndexBar

## Basic usage

```tsx preview
import { IndexBar } from '@koi-ui/core';
const groups = [
  { index: 'A', items: ['Apple', 'Ant', 'Avocado'] },
  { index: 'B', items: ['Banana', 'Blueberry'] },
  { index: 'C', items: ['Cherry', 'Coconut'] },
  { index: 'D', items: ['Date', 'Dragonfruit'] },
];
export default () => (
  <div className="koi-indexbar-demo flex h-[20rem] min-h-0 overflow-hidden rounded-md border border-border">
    <IndexBar className="min-h-0 min-w-0 flex-1" groups={groups} renderItem={(item) => <span>{item}</span>} />
  </div>
);
```

## stickyOffset & onIndexChange

```tsx preview
import { IndexBar, Text } from '@koi-ui/core';
import { useState } from 'react';
const groups = [
  { index: 'A', title: 'Group A', items: ['Apple', 'Ant', 'Avocado', 'Apricot'] },
  { index: 'B', title: 'Group B', items: ['Banana', 'Blueberry', 'Blackberry'] },
  { index: 'C', title: 'Group C', items: ['Cherry', 'Coconut', 'Cranberry'] },
  { index: 'D', title: 'Group D', items: ['Date', 'Dragonfruit', 'Durian'] },
];
export default () => {
  const [active, setActive] = useState('A');
  return (
    <div className="space-y-2">
      <Text muted size="sm">Active index: {active}</Text>
      <div className="koi-indexbar-demo flex h-[20rem] min-h-0 overflow-hidden rounded-md border border-border">
        <IndexBar
          className="min-h-0 min-w-0 flex-1"
          groups={groups}
          stickyOffset={8}
          onIndexChange={setActive}
          renderItem={(item) => <span>{item}</span>}
        />
      </div>
    </div>
  );
};
```
