# PullToRefresh

## Basic usage

```tsx preview
import { PullToRefresh, List } from '@koi-ui/core';
export default () => <PullToRefresh onRefresh={async()=>{}}><List items={[{key:'1',title:'Item 1'},{key:'2',title:'Item 2'}]} /></PullToRefresh>;
```
