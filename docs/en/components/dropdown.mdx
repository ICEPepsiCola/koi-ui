# Dropdown

## Basic usage

```tsx preview
import { Dropdown, Button } from '@koi-ui/core';
export default () => (
  <Dropdown trigger={<Button>Dropdown</Button>} items={[{ key: '1', label: 'Option 1' }, { key: '2', label: 'Option 2' }]} />
);
```

## Disabled items

```tsx preview
import { Dropdown, Button } from '@koi-ui/core';
export default () => (
  <Dropdown
    trigger={<Button>Actions</Button>}
    items={[
      { key: 'edit', label: 'Edit' },
      { key: 'copy', label: 'Copy' },
      { key: 'delete', label: 'Delete', disabled: true },
    ]}
  />
);
```
