# Dropdown

## Basic usage

```tsx preview
import { Dropdown, Button } from '@koi-ui/core';
export default () => (
  <Dropdown
    trigger={<Button>Dropdown</Button>}
    items={[
      { key: '1', label: 'Option 1' },
      { key: '2', label: 'Option 2' },
    ]}
  />
);
```

## Disabled items

```tsx preview
import { Dropdown, Button } from '@koi-ui/core';
export default () => (
  <Dropdown
    trigger={<Button>Actions</Button>}
    items={[
      { key: 'edit', label: 'Edit' },
      { key: 'copy', label: 'Copy' },
      { key: 'delete', label: 'Delete', disabled: true },
    ]}
  />
);
```

## Placement

```tsx preview
import { Dropdown, Button, Space } from '@koi-ui/core';
const items = [
  { key: '1', label: 'Option 1' },
  { key: '2', label: 'Option 2' },
];
export default () => (
  <Space wrap>
    <Dropdown trigger={<Button>bottom-start</Button>} items={items} placement="bottom-start" />
    <Dropdown trigger={<Button>bottom-end</Button>} items={items} placement="bottom-end" />
    <Dropdown trigger={<Button>top-start</Button>} items={items} placement="top-start" />
    <Dropdown trigger={<Button>top-end</Button>} items={items} placement="top-end" />
  </Space>
);
```

## Title (mobile)

```tsx preview
import { Dropdown, Button } from '@koi-ui/core';
export default () => (
  <Dropdown
    title="Actions"
    trigger={<Button>With title</Button>}
    items={[
      { key: 'edit', label: 'Edit' },
      { key: 'share', label: 'Share' },
    ]}
  />
);
```

## Disabled

```tsx preview
import { Dropdown, Button } from '@koi-ui/core';
export default () => (
  <Dropdown
    disabled
    trigger={<Button>Disabled</Button>}
    items={[
      { key: '1', label: 'Option 1' },
      { key: '2', label: 'Option 2' },
    ]}
  />
);
```
