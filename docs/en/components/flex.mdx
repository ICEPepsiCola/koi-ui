
# Flex

## Basic usage

```tsx preview
import { Flex, Button } from '@koi-ui/core';
export default () => (
  <Flex justify="between" align="center" className="rounded-md border border-border p-4">
    <span>Left</span>
    <Button size="sm">Action</Button>
  </Flex>
);
```

## Direction

```tsx preview
import { Flex, Tag } from '@koi-ui/core';
export default () => (
  <div className="space-y-4">
    <Flex gap={2}>
      <Tag>row</Tag>
      <Tag color="secondary">default</Tag>
    </Flex>
    <Flex direction="column" gap={2}>
      <Tag>column</Tag>
      <Tag color="secondary">vertical</Tag>
    </Flex>
    <Flex vertical gap={2}>
      <Tag>vertical</Tag>
      <Tag color="secondary">shorthand</Tag>
    </Flex>
  </div>
);
```

## Align & justify

```tsx preview
import { Flex, Tag } from '@koi-ui/core';
export default () => (
  <div className="space-y-4">
    <Flex justify="between" align="center" className="h-16 rounded-md border border-border px-4">
      <Tag>Left</Tag>
      <Tag color="success">Right</Tag>
    </Flex>
    <Flex justify="center" gap={2} className="rounded-md border border-border p-3">
      <Tag>center</Tag>
      <Tag color="secondary">center</Tag>
    </Flex>
    <Flex justify="around" gap={2} className="rounded-md border border-border p-3">
      <Tag>around</Tag>
      <Tag color="secondary">around</Tag>
      <Tag variant="outline">around</Tag>
    </Flex>
  </div>
);
```

## Gap & wrap

```tsx preview
import { Flex, Tag } from '@koi-ui/core';
export default () => (
  <div className="space-y-4">
    <Flex gap={4}>
      <Tag>gap 4</Tag>
      <Tag color="secondary">gap 4</Tag>
      <Tag variant="outline">gap 4</Tag>
    </Flex>
    <Flex wrap="wrap" gap={2} className="max-w-48">
      <Tag>A</Tag>
      <Tag>B</Tag>
      <Tag>C</Tag>
      <Tag>D</Tag>
      <Tag>E</Tag>
      <Tag>F</Tag>
    </Flex>
  </div>
);
```

## Responsive

```tsx preview
import { Flex, Tag } from '@koi-ui/core';
export default () => (
  <Flex responsive gap={2} className="rounded-md border border-border p-3">
    <Tag>column on narrow</Tag>
    <Tag color="secondary">row from md+</Tag>
  </Flex>
);
```
