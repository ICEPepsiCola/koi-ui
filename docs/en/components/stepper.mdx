# Stepper

## Basic usage

```tsx preview
import { Stepper } from '@koi-ui/core';
export default () => <Stepper defaultValue={1} />;
```

## Size

```tsx preview
import { Stepper, Space } from '@koi-ui/core';
export default () => (
  <Space align="center" size="middle">
    <Stepper size="sm" defaultValue={1} />
    <Stepper size="md" defaultValue={2} />
    <Stepper size="lg" defaultValue={3} />
  </Space>
);
```

## Colors

```tsx preview
import { Stepper, Space } from '@koi-ui/core';
export default () => (
  <Space wrap size="middle">
    <Stepper color="primary" defaultValue={1} />
    <Stepper color="success" defaultValue={2} />
    <Stepper color="warning" defaultValue={3} />
    <Stepper color="error" defaultValue={4} />
  </Space>
);
```

## Min / max / step

```tsx preview
import { Stepper } from '@koi-ui/core';
export default () => <Stepper min={0} max={10} step={0.5} defaultValue={2} />;
```

## Label

```tsx preview
import { Stepper } from '@koi-ui/core';
export default () => <Stepper label="Quantity" min={1} max={99} defaultValue={1} />;
```

## Disabled

```tsx preview
import { Stepper } from '@koi-ui/core';
export default () => <Stepper disabled defaultValue={5} />;
```

## Controlled

```tsx preview
import { Stepper, Space, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [value, setValue] = useState(3);
  return (
    <Space align="center" size="middle">
      <Stepper value={value} onChange={setValue} min={0} max={20} />
      <Text size="sm" muted>Value: {value}</Text>
    </Space>
  );
};
```
