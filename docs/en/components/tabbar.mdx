# TabBar

## Basic usage

```tsx preview
import { TabBar } from '@koi-ui/core';
export default () => (
  <TabBar
    defaultActiveKey="home"
    items={[
      { key: 'home', label: 'Home' },
      { key: 'discover', label: 'Discover' },
      { key: 'mine', label: 'Me' },
    ]}
  />
);
```

## Icons & badges

```tsx preview
import { TabBar, Icon } from '@koi-ui/core';
export default () => (
  <TabBar
    defaultActiveKey="home"
    items={[
      { key: 'home', label: 'Home', icon: <Icon name="check" size="sm" /> },
      { key: 'search', label: 'Search', icon: <Icon name="search" size="sm" />, badge: 3 },
      { key: 'mine', label: 'Me', icon: <Icon name="plus" size="sm" />, badge: 'New' },
    ]}
  />
);
```

## Disabled items

```tsx preview
import { TabBar, Icon } from '@koi-ui/core';
export default () => (

  />

```

## Controlled

```tsx preview
import { TabBar, Icon, Space, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [activeKey, setActiveKey] = useState('home');
  return (
    <Space direction="vertical" size="middle" className="w-full items-stretch">
      <Text size="sm" muted>activeKey: {activeKey}</Text>
      <TabBar
        activeKey={activeKey}
        onChange={setActiveKey}
        items={[
          { key: 'home', label: 'Home', icon: <Icon name="check" size="sm" /> },
          { key: 'search', label: 'Search', icon: <Icon name="search" size="sm" /> },
          { key: 'mine', label: 'Me', icon: <Icon name="plus" size="sm" /> },
        ]}
      />
    </Space>
  );
};
```

## Safe area

```tsx preview
import { TabBar, Icon, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full items-stretch">
    <TabBar
      safeArea
      defaultActiveKey="home"
      items={[
        { key: 'home', label: 'On', icon: <Icon name="check" size="sm" /> },
        { key: 'mine', label: 'Me', icon: <Icon name="plus" size="sm" /> },
      ]}
    />
    <TabBar
      safeArea={false}
      defaultActiveKey="home"
      items={[
        { key: 'home', label: 'Off', icon: <Icon name="check" size="sm" /> },
        { key: 'mine', label: 'Me', icon: <Icon name="plus" size="sm" /> },
      ]}
    />
  </Space>
);
```
