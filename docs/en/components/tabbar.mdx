## Basic usage

```tsx preview
import { TabBar } from '@koi-ui/core';
import { useState } from 'react';
export default () => { const [k,setK]=useState('home'); return <TabBar activeKey={k} onChange={setK} items={[{ key:'home', label:'Home' }, { key:'mine', label:'Me' }]} />; };
```
