
# List

## Basic usage

```tsx preview
import { List } from '@koi-ui/core';
export default () => (
  <List
    items={[
      { key: '1', title: 'Item one', description: 'Description one' },
      { key: '2', title: 'Item two', description: 'Description two' },
    ]}
  />
);
```

## Bordered & size

```tsx preview
import { List, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <List
      bordered
      size="sm"
      items={[
        { key: '1', title: 'Small list', description: 'size=sm' },
        { key: '2', title: 'Another item', description: 'Desc' },
      ]}
    />
    <List
      bordered
      size="lg"
      items={[
        { key: '1', title: 'Large list', description: 'size=lg' },
        { key: '2', title: 'Another item', description: 'Desc' },
      ]}
    />
  </Space>
);
```

## Avatar & extras

```tsx preview
import { List, Avatar, Button } from '@koi-ui/core';
export default () => (
  <List
    bordered
    header="Members"
    footer="2 people"
    items={[
      {
        key: '1',
        title: 'Alice',
        description: 'Product',
        avatar: <Avatar size="sm" fallback="A" />,
        extra: 'Online',
        actions: <Button size="sm" variant="ghost">View</Button>,
      },
      {
        key: '2',
        title: 'Bob',
        description: 'Design',
        avatar: <Avatar size="sm" fallback="B" />,
        extra: 'Busy',
      },
    ]}
  />
);
```
