# Table

## Basic usage

```tsx preview
import { Table } from '@koi-ui/core';
export default () => <Table columns={[{key:'name',title:'Name'},{key:'age',title:'Age'}]} data={[{name:'Alice',age:'20'},{name:'Bob',age:'25'}]} />;
```

## Empty data

```tsx preview
import { Table } from '@koi-ui/core';
export default () => (
  <Table
    columns={[{ key: 'name', title: 'Name' }, { key: 'role', title: 'Role' }]}
    data={[]}
    emptyText="No members"
  />
);
```
