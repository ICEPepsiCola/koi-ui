# Swiper

## Basic usage

```tsx preview
import { Swiper } from '@koi-ui/core';
export default () => (
  <Swiper>
    <div className="flex h-32 items-center justify-center bg-muted text-lg font-medium">Slide 1</div>
    <div className="flex h-32 items-center justify-center bg-primary/10 text-lg font-medium">Slide 2</div>
    <div className="flex h-32 items-center justify-center bg-muted/80 text-lg font-medium">Slide 3</div>
    <div className="flex h-32 items-center justify-center bg-primary/20 text-lg font-medium">Slide 4</div>
  </Swiper>
);
```

## Arrow navigation

```tsx preview
import { Swiper } from '@koi-ui/core';
export default () => (
  <Swiper showArrows>
    <div className="flex h-32 items-center justify-center bg-muted text-lg font-medium">Slide 1</div>
    <div className="flex h-32 items-center justify-center bg-primary/10 text-lg font-medium">Slide 2</div>
    <div className="flex h-32 items-center justify-center bg-muted/80 text-lg font-medium">Slide 3</div>
    <div className="flex h-32 items-center justify-center bg-primary/20 text-lg font-medium">Slide 4</div>
  </Swiper>
);
```

## Loop, autoplay & dots

```tsx preview
import { Swiper, Space, Text } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="middle" className="w-full">
    <div>
      <Text muted size="sm" className="mb-2 block">loop + autoplay</Text>
      <Swiper loop autoplay interval={2500}>
        <div className="flex h-28 items-center justify-center bg-muted font-medium">A</div>
        <div className="flex h-28 items-center justify-center bg-primary/10 font-medium">B</div>
        <div className="flex h-28 items-center justify-center bg-primary/20 font-medium">C</div>
      </Swiper>
    </div>
    <div>
      <Text muted size="sm" className="mb-2 block">showDots=false · loop=false</Text>
      <Swiper loop={false} showDots={false} showArrows>
        <div className="flex h-28 items-center justify-center bg-muted font-medium">1</div>
        <div className="flex h-28 items-center justify-center bg-primary/10 font-medium">2</div>
        <div className="flex h-28 items-center justify-center bg-primary/20 font-medium">3</div>
      </Swiper>
    </div>
  </Space>
);
```
