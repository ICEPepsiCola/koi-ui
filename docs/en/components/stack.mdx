
# Stack

## Basic usage

```tsx preview
import { Stack, Text } from '@koi-ui/core';
export default () => (
  <Stack gap={2}>
    <Text weight="bold">Title</Text>
    <Text muted>Description</Text>
  </Stack>
);
```

## Direction & gap

```tsx preview
import { Stack, Tag } from '@koi-ui/core';
export default () => (
  <div className="space-y-4">
    <Stack direction="col" gap={2}>
      <Tag>col gap 2</Tag>
      <Tag variant="secondary">column</Tag>
    </Stack>
    <Stack direction="row" gap={4}>
      <Tag>row gap 4</Tag>
      <Tag variant="secondary">row</Tag>
      <Tag variant="outline">row</Tag>
    </Stack>
  </div>
);
```

## Align & justify

```tsx preview
import { Stack, Tag } from '@koi-ui/core';
export default () => (
  <div className="space-y-4">
    <Stack direction="row" align="center" justify="between" className="rounded-md border border-border p-3">
      <Tag>Left</Tag>
      <Tag variant="success">Right</Tag>
    </Stack>
    <Stack direction="row" align="end" gap={2} className="h-16 rounded-md border border-border p-3">
      <Tag>end</Tag>
      <Tag variant="secondary">align end</Tag>
    </Stack>
  </div>
);
```

## Responsive

```tsx preview
import { Stack, Tag } from '@koi-ui/core';
export default () => (
  <Stack responsive gap={2} className="rounded-md border border-border p-3">
    <Tag>column on narrow</Tag>
    <Tag variant="secondary">row from md+</Tag>
  </Stack>
);
```
