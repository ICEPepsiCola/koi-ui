# Stack

## Basic usage

```tsx preview
import { Stack, Text } from '@koi-ui/core';
export default () => (
  <Stack gap={2}>
    <Text weight="bold">Title</Text>
    <Text muted>Description</Text>
  </Stack>
);
```
