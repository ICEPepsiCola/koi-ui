# Notification

## Basic usage

```tsx preview
import { Notification, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>Show notification</Button>
      <Notification
        open={open}
        onClose={() => setOpen(false)}
        title="Notice"
        description="This is a notification message"
      />
    </>
  );
};
```

## Colors

Semantic colors aligned with Button: `info` | `success` | `warning` | `error`.

```tsx preview
import { Notification, Button, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [color, setColor] = useState(null);
  return (
    <>
      <Space wrap>
        <Button variant="outline" onClick={() => setColor('info')}>info</Button>
        <Button variant="soft" onClick={() => setColor('success')}>success</Button>
        <Button color="secondary" onClick={() => setColor('warning')}>warning</Button>
        <Button color="error" onClick={() => setColor('error')}>error</Button>
      </Space>
      {color ? (
        <Notification
          color={color}
          open
          onClose={() => setColor(null)}
          title={color}
          description={`${color} notification`}
        />
      ) : null}
    </>
  );
};
```

## Placement

```tsx preview
import { Notification, Button, Space } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [placement, setPlacement] = useState(null);
  return (
    <>
      <Space wrap>
        <Button size="sm" onClick={() => setPlacement('topRight')}>topRight</Button>
        <Button size="sm" variant="outline" onClick={() => setPlacement('topLeft')}>topLeft</Button>
        <Button size="sm" color="secondary" onClick={() => setPlacement('bottomRight')}>bottomRight</Button>
        <Button size="sm" variant="ghost" onClick={() => setPlacement('bottomLeft')}>bottomLeft</Button>
      </Space>
      {placement ? (
        <Notification
          open
          placement={placement}
          onClose={() => setPlacement(null)}
          title="Placement demo"
          description={`placement=${placement}`}
        />
      ) : null}
    </>
  );
};
```

## useNotification

```tsx preview
import { NotificationProvider, useNotification, Button, Space } from '@koi-ui/core';

function Demo() {
  const { notify } = useNotification();
  return (
    <Space wrap>
      <Button onClick={() =>
          notify({
            title: 'Saved',
            description: 'Your settings were updated',
            color: 'success',
          })
        }
      >
        Success
      </Button>
      <Button variant="outline"onClick={() =>
          notify({
            title: 'Warning',
            description: 'Check your network connection',
            color: 'warning',
            placement: 'bottomRight',
          })
        }
      >
        Warning (bottomRight)
      </Button>
    </Space>
  );
}

export default () => (
  <NotificationProvider>
    <Demo />
  </NotificationProvider>
);
```
