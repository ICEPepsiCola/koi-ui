
# Steps

## Basic usage

```tsx preview
import { Steps } from '@koi-ui/core';
export default () => (
  <Steps
    current={1}
    items={[
      { title: 'Done' },
      { title: 'In progress' },
      { title: 'Waiting' },
    ]}
  />
);
```

## Finish state & description

```tsx preview
import { Steps } from '@koi-ui/core';
export default () => (
  <Steps
    current={3}
    items={[
      { title: 'Details', description: 'Done' },
      { title: 'Confirm', description: 'Done' },
      { title: 'Pay', description: 'Done' },
      { title: 'Complete' },
    ]}
  />
);
```

## Vertical & size

```tsx preview
import { Steps, Space } from '@koi-ui/core';
export default () => (
  <Space direction="vertical" size="large" className="w-full">
    <Steps
      direction="vertical"
      current={1}
      items={[
        { title: 'Submit', description: 'Fill in the form' },
        { title: 'Review', description: 'About 1 business day' },
        { title: 'Done' },
      ]}
    />
    <Steps
      size="sm"
      current={0}
      items={[{ title: 'sm' }, { title: 'sm' }, { title: 'sm' }]}
    />
  </Space>
);
```
