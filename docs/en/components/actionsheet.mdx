# ActionSheet

## Basic usage

```tsx preview
import { ActionSheet, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>Open</Button>
      <ActionSheet
        open={open}
        onClose={() => setOpen(false)}
        actions={[
          { key: '1', text: 'Option 1' },
          { key: '2', text: 'Option 2' },
        ]}
      />
    </>
  );
};
```

## Title and description

```tsx preview
import { ActionSheet, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setOpen(true)}>More</Button>
      <ActionSheet
        open={open}
        onClose={() => setOpen(false)}
        title="Share to"
        description="Pick a platform"
        actions={[
          { key: 'wechat', text: 'WeChat' },
          { key: 'link', text: 'Copy link' },
        ]}
      />
    </>
  );
};
```

## Destructive & disabled

```tsx preview
import { ActionSheet, Button } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button color="error" onClick={() => setOpen(true)}>
        Manage
      </Button>
      <ActionSheet
        open={open}
        onClose={() => setOpen(false)}
        title="Actions"
        cancelText="Close"
        actions={[
          { key: 'edit', text: 'Edit' },
          { key: 'archive', text: 'Archive', disabled: true },
          { key: 'delete', text: 'Delete', destructive: true },
        ]}
      />
    </>
  );
};
```
