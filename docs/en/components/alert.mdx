# Alert

## Basic usage

```tsx preview
import { Alert } from '@koi-ui/core';
export default () => <Alert variant="info" title="Info" description="This is an info message" />;
```

## Variants

```tsx preview
import { Alert, Stack } from '@koi-ui/core';
export default () => (
  <Stack gap={3}>
    <Alert variant="info" title="Info" description="General tip" />
    <Alert variant="success" title="Success" description="Done" />
    <Alert variant="warning" title="Warning" description="Please check input" />
    <Alert variant="error" title="Error" description="Submit failed" />
  </Stack>
);
```
