# Alert

## Basic usage

```tsx preview
import { Alert } from '@koi-ui/core';
export default () => (
  <Alert color="info" title="Info" description="This is an info message" />
);
```

## Colors

```tsx preview
import { Alert, Stack } from '@koi-ui/core';
export default () => (
  <Stack gap={3}>
    <Alert color="neutral" title="Neutral" description="Neutral tip" />
    <Alert color="info" title="Info" description="General tip" />
    <Alert color="success" title="Success" description="Done" />
    <Alert color="warning" title="Warning" description="Please check input" />
    <Alert color="error" title="Error" description="Submit failed" />
  </Stack>
);
```

## Variants

```tsx preview
import { Alert, Stack } from '@koi-ui/core';
export default () => (
  <Stack gap={3}>
    <Alert color="info" variant="soft" title="Soft" description="Soft fill (default)" />
    <Alert color="success" variant="outline" title="Outline" description="Outline" />
    <Alert color="warning" variant="dash" title="Dash" description="Dashed outline" />
    <Alert color="error" variant="solid" title="Solid" description="Solid" />
  </Stack>
);
```

## Actions & responsive

```tsx preview
import { Alert, Button, Stack } from '@koi-ui/core';
export default () => (
  <Stack gap={3}>
    <Alert
      responsive
      color="info"
      title="New message"
      description="You have 1 unread message"
      action={<Button size="sm">See</Button>}
    />
    <Alert
      responsive
      color="warning"
      description="We use cookies for no reason"
      action={
        <>
          <Button size="sm" variant="ghost">Deny</Button>
          <Button size="sm" color="primary">Accept</Button>
        </>
      }
    />
  </Stack>
);
```

## Closable

```tsx preview
import { Alert, Stack } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [visible, setVisible] = useState(true);
  return (
    <Stack gap={3}>
      {visible ? (
        <Alert
          closable
          color="info"
          title="Closable"
          description="Click the close icon"
          onClose={() => setVisible(false)}
        />
      ) : null}
      <Alert
        closable
        color="warning"
        title="closable only"
        description="Shows the close button; without onClose, click is a no-op"
        icon={null}
      />
    </Stack>
  );
};
```
