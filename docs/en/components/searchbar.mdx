
# SearchBar

## Basic usage

```tsx preview
import { SearchBar } from '@koi-ui/core';
export default () => <SearchBar placeholder="Search" />;
```

## Controlled search

```tsx preview
import { SearchBar, Text } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [keyword, setKeyword] = useState('');
  return (
    <div>
      <SearchBar value={keyword} onChange={setKeyword} placeholder="Search products" />
      <Text muted className="mt-2 block text-sm">Keyword: {keyword || 'None'}</Text>
    </div>
  );
};
```

## clearable

```tsx preview
import { SearchBar } from '@koi-ui/core';
import { useState } from 'react';
export default () => {
  const [keyword, setKeyword] = useState('耳机');
  return <SearchBar value={keyword} onChange={setKeyword} clearable placeholder="Search products" />;
};
```

## Alongside other fields

```tsx preview
import { Cascader, Input, SearchBar, Stack } from '@koi-ui/core';

const options = [
  { label: 'Zhejiang', value: 'zj', children: [{ label: 'Hangzhou', value: 'hz' }] },
];

export default () => (
  <Stack gap={3} className="max-w-md w-full">
    <SearchBar placeholder="Search products" />
    <Cascader options={options} placeholder="Select region" />
    <Input placeholder="备注" />
  </Stack>
);
```
