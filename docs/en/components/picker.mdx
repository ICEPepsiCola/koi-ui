# Picker

## Basic usage

```tsx preview
import { Picker } from '@koi-ui/core';
export default () => (
  <Picker
    placeholder="Select"
    columns={[
      { options: [{ label: 'Mon', value: '1' }, { label: 'Tue', value: '2' }, { label: 'Wed', value: '3' }] },
      { options: [{ label: 'AM', value: 'am' }, { label: 'PM', value: 'pm' }] },
    ]}
  />
);
```

## clearable

```tsx preview
import { Picker } from '@koi-ui/core';
import { useState } from 'react';
const columns = [
  { options: [{ label: 'Mon', value: '1' }, { label: 'Tue', value: '2' }, { label: 'Wed', value: '3' }] },
  { options: [{ label: 'AM', value: 'am' }, { label: 'PM', value: 'pm' }] },
];
export default () => {
  const [value, setValue] = useState(['2', 'pm']);
  return <Picker value={value} onChange={setValue} columns={columns} clearable placeholder="Select time段" />;
};
```
