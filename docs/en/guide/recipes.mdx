## Sign in

```tsx preview
import {
  Button,
  Card,
  Checkbox,
  Form,
  Input,
  Space,
  Stack,
  Text,
  toast,
} from '@koi-ui/core';

export default () => (
  <Card className="max-w-md w-full">
    <Stack gap={4}>
      <div>
        <Text className="text-lg font-semibold">Welcome back</Text>
        <Text className="text-sm text-muted-foreground">Sign in to continue</Text>
      </div>
      <Form
        layout="vertical"
        onFinish={() => toast.success('Signed in')}
        onFinishFailed={() => toast.error('Check the form')}
      >
        <Form.Item
          name="email"
          label="Email"
          rules={[
            { required: true, message: 'Email is required' },
            { type: 'email', message: 'Enter a valid email' },
          ]}
        >
          <Input placeholder="name@company.com" clearable />
        </Form.Item>
        <Form.Item
          name="password"
          label="Password"
          rules={[{ required: true, min: 6, message: 'At least 6 characters' }]}
        >
          <Input type="password" placeholder="••••••••" clearable />
        </Form.Item>
        <Form.Item name="remember" valuePropName="checked">
          <Checkbox label="Remember me" />
        </Form.Item>
        <Space>
          <Button type="submit" variant="primary">Sign in</Button>
          <Button type="button" variant="ghost">Create account</Button>
        </Space>
      </Form>
    </Stack>
  </Card>
);
```

## Data list

```tsx preview
import { useState } from 'react';
import {
  Badge,
  Button,
  SearchBar,
  Space,
  Stack,
  Table,
  Tag,
} from '@koi-ui/core';

const rows = [
  { id: '1', name: 'North Harbor', status: 'active', owners: 12 },
  { id: '2', name: 'Cedar Labs', status: 'trial', owners: 4 },
  { id: '3', name: 'Moonlight Co', status: 'paused', owners: 7 },
];

export default () => {
  const [query, setQuery] = useState('');
  const data = rows.filter((row) =>
    row.name.toLowerCase().includes(query.trim().toLowerCase()),
  );

  return (
    <Stack gap={4} className="w-full">
      <Space className="w-full justify-between" wrap>
        <SearchBar className="min-w-0 flex-1 max-w-md" value={query} onChange={setQuery} placeholder="Filter teams" />
        <Button variant="soft">New team</Button>
      </Space>
      <Table
        columns={[
          { key: 'name', title: 'Team' },
          {
            key: 'status',
            title: 'Status',
            render: (value) =>
              value === 'active' ? (
                <Tag variant="success">Active</Tag>
              ) : value === 'trial' ? (
                <Tag variant="warning">Trial</Tag>
              ) : (
                <Tag variant="ghost">Paused</Tag>
              ),
          },
          {
            key: 'owners',
            title: 'Members',
            render: (value) => <Badge count={Number(value)} variant="soft" />,
          },
        ]}
        data={data}
      />
    </Stack>
  );
};
```

## Settings form

```tsx preview
import {
  Button,
  Cascader,
  Form,
  Input,
  Select,
  Space,
  Stack,
  Switch,
  Text,
  toast,
} from '@koi-ui/core';

export default () => {
  const [form] = Form.useForm();
  return (
    <Stack gap={4} className="max-w-lg w-full">
      <div>
        <Text className="text-lg font-semibold">Workspace settings</Text>
        <Text className="text-sm text-muted-foreground">
          Adaptive form layout — labels stack on mobile.
        </Text>
      </div>
      <Form
        form={form}
        initialValues={{ name: 'Koi Studio', plan: 'pro', notify: true }}
        onFinish={() => toast.success('Saved')}
      >
        <Form.Item
          name="name"
          label="Workspace name"
          rules={[{ required: true, message: 'Required' }]}
        >
          <Input clearable />
        </Form.Item>
        <Form.Item name="plan" label="Plan">
          <Select
            options={[
              { value: 'free', label: 'Free' },
              { value: 'pro', label: 'Pro' },
              { value: 'team', label: 'Team' },
            ]}
          />
        </Form.Item>
        <Form.Item name="region" label="Region">
          <Cascader
            options={[
              {
                label: 'East',
                value: 'east',
                children: [{ label: 'Shanghai', value: 'sh' }],
              },
              {
                label: 'North',
                value: 'north',
                children: [{ label: 'Beijing', value: 'bj' }],
              },
            ]}
          />
        </Form.Item>
        <Form.Item name="notify" label="Email alerts" valuePropName="checked">
          <Switch />
        </Form.Item>
        <Space>
          <Button type="submit" variant="primary">Save</Button>
          <Button type="button" variant="outline" onClick={() => form.resetFields()}>
            Reset
          </Button>
        </Space>
      </Form>
    </Stack>
  );
};
```
